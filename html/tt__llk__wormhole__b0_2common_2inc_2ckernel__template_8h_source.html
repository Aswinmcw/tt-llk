14  // Here is the basic outline of a MOP loop and the definition of the
15  // variables used.
16  // LOOP_OUTER: <OUTER_LOOP_COUNT>
17  // START_OP
18  // LOOP_INNER: <INNER_LOOP_COUNT>
19  // LOOP_OP0
20  // LOOP_OP1
21  // END_LOOP_INNER
22  // END_OP0
23  // END_OP1
24  // END_LOOP_OUTER
25 
26  const uint m_outer_loop_len;
27  const uint m_inner_loop_len;
28  uint m_loop_op0;
29  uint m_loop_op1;
30  uint m_end_op0, m_end_op1, m_start_op0;
31  uint m_loop0_last_instr; // In the last iteration of the outer loop, this instruction replaces the inner loop instruction, if constructed with one inner
32  // loop instruction or the second inner loop instruction, if constructed with two inner loop instructions (see below example).
33  uint m_loop1_last_instr; // In the last iteration of the inner loop, this instruction replaces the inner loop instruction, if constructed with one inner
34  // loop instruction or the second inner loop instruction, if constructed with two inner loop instructions (see below example).
35 
36  // Note: The last iteration of inner loop will also be the last iteration of the outer loop when outer loop length = 1.
37  // This means that in this case, last_inner_loop_instr will be replaced by the last_outer_loop_instr
38  // NOTE:
39  // This is how m_loop0_last_instr and m_loop1_last_instr are executed:
40  //
41  // if(last_inner_loop_iter && last_outer_loop_iter) m_loop_op1 = m_loop0_last_instr;
42  // else if(last_inner_loop_iter) m_loop_op1 = m_loop1_last_instr;
43  // else m_loop_op1 = m_loop_op1;
44 
45 public:
46  ckernel_template(uint outer_loop_len, uint inner_loop_len, uint loop_op);

63  // Unpack template is a single-loop template that allows for some dynamic selection of instructions based on zmask
64  // The basic template looks like this:
65  //
66  // LOOP:
67  // if (zmask[iteration]):
68  // UNPACR_A0
69  // UNPACR_A1
70  // UNPACR_A2
71  // UNPACR_A3
72  // UNPACR_B
73  // else
74  // SKIP_A0
75  // SKIP_A1
76  // SKIP_A2
77  // SKIP_A3
78  // SKIP_B
79  //
80  // The configuration allows the following changes:
81  // - B enable - if 0, removes the UNPACR_B/SKIP_B instruction
82  // - HALO enable - if 0, removes A1/2/3 instructions
83  // - Each of the UNPACR/SKIP instructions can be anything at all, although the most common use is for UNPACR to unpack, and SKIP to be an
84  // unpack NOP that increments the context counter

343 inline void ckernel_unpack_template::run(volatile uint *instrn_buffer, const uint8_t count, const uint32_t zmask)
344 {
345  FWASSERT("Unpack template only supports loops up to 128", count <= 128);
346  TT_MOP_CFG(zmask >> 16); // Set the top 16 bits of zmask - we could skip this for count <= 16
347  TT_MOP(0, count - 1, zmask & 0xFFFF); // Run the template
348 }
