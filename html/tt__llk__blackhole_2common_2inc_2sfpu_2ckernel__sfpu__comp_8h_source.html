24 template <bool APPROXIMATION_MODE, bool invert_output, bool check_zero, bool second_check, bool is_less_than_equal_zero, int ITERATIONS>
25 inline void _calculate_comp_(const int iterations, uint exponent_size_8)
26 {
27  // output_0 and output_1 hold the outputs use use when a zero or negative check is true/false.
28  // False = 0.0 = kCONST_0 (5/8-bit exponent format)
29  // True = 1.0 = kCONST_1_FP16B (8-bit exponent format)
30  // SFPU uses 8-bit exponent in operations so loading these constants in 8-bit exponent format.
31  // Although a command flag can tell SFPU to re-bias a 5-bit exponent to 8-bit, we are loading 8-bit
32  // exponent and telling SFPU to not add any bias to these constants.
33  constexpr float output_0 = invert_output ? 0.0f : 1.0f;
34  constexpr float output_1 = invert_output ? 1.0f : 0.0f;
35 
36  for (int d = 0; d < iterations; d++)
37  {
38  sfpi::vFloat v = sfpi::dst_reg[0];
39  sfpi::vFloat flag1, flag2;
40  if constexpr (check_zero)
41  {
42  v_if (_sfpu_is_fp16_zero_(v, exponent_size_8))
43  {
44  _calculate_comp_init_flag_(second_check, flag1, flag2, output_0);
45  }
46  v_else
47  {
48  _calculate_comp_init_flag_(second_check, flag1, flag2, output_1);
49  }
50  v_endif;
51  }
52  else
53  {
54  v_if (v < 0.0F)
55  {
56  _calculate_comp_init_flag_(second_check, flag1, flag2, output_0);
57  }
58  v_else
59  {
60  _calculate_comp_init_flag_(second_check, flag1, flag2, output_1);
61  }
62  v_endif;
63  }
64 
65  sfpi::vFloat result;
66  if constexpr (second_check)
67  {
68  // less_than_equal_zero
69  // flag1 = 0x3F80(1.0) if DST < 0 else 0
70  // flag2 = 0x3F80(1.0) if DST == 0 else 0
71  // Do a bitwise Or (flag1 | flag2) to get <= condition.
72  // flag1 < 0 OR flag2 == 0 => DST is Less than or Equal to zero.
73  // Result will be either 0x0000(0.0) or 0x3F80(1.0)
74  if constexpr (is_less_than_equal_zero)
75  {
76  result = sfpi::reinterpret<sfpi::vFloat>(sfpi::reinterpret<sfpi::vUInt>(flag1) | sfpi::reinterpret<sfpi::vUInt>(flag2));
77  }
78  else
79  {
80  // greater_than_zero
81  // flag1 = 0x3F80(1.0) if DST >= 0 else 0
82  // flag2 = 0x3F80(1.0) if DST != 0 else 0
83  // Do a bitwise And (flag1 & flag2) to get > condition.
84  // flag2 >= 0 AND flag1 != 0 => DST is Greater than zero
85  // Result will be either 0x0000(0.0) or 0x3F80(1.0)
86  result = sfpi::reinterpret<sfpi::vFloat>(sfpi::reinterpret<sfpi::vUInt>(flag1) & sfpi::reinterpret<sfpi::vUInt>(flag2));
87  }
88  }
89  else
90  {
91  result = flag1;
92  }
93 
94  sfpi::dst_reg[0] = result;
95 
96  sfpi::dst_reg++;
97  }
98 }
