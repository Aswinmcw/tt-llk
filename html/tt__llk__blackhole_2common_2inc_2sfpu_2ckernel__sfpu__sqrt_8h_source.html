30

33  sfpi::dst_reg[0] = sfpi::reinterpret<sfpi::vFloat>(val_s);
34  }
35  else

40  v_if (val != 0.0f)
41  {
42  sfpi::vUInt magic = sfpi::vConstIntPrgm0;

44 
45  // Reciproot iterations
46  for (int r = 0; r < RECIPROCAL_ITERATIONS; r++)
47  {
48  // x*r*(1.5f - xhalf*r*r);
49  approx = ((approx * approx) * (val * -0.5f) + 1.5f) * approx;
50  }
51 
52  sfpi::dst_reg[0] = approx * val;
53  }
