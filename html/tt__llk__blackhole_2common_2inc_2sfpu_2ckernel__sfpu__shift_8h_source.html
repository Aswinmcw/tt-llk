42 {

50  TTI_SFPMOV(0, 0, 4, 0); // save shift_value for later
51  // shift right
52  TTI_SFPIADD(0, 9, 1, 6); // take negative of shift_amount to shift right
53  TTI_SFPSHFT(0, 1, 0, 0);
54  // if shift_value was negative, need to shift in 1's manually
55  TTI_SFPSETCC(0, 4, 0, 0); // only run if shift_value is negative
56  TTI_SFPSETCC(0, 1, 0, 2); // only needed if shift_amount>0
57  TTI_SFPIADD(0x020, 1, 2, 5); // take 32-shift_amount (0x020 = 32)
58  TTI_SFPNOT(0, 9, 3, 0); // put all 1's into LREG3
59  TTI_SFPSHFT(0, 2, 3, 0); // shift all 1's by 32-shift_amount
60  TTI_SFPOR(0, 3, 0, 0); // OR in the 1's
61  TTI_SFPENCC(0, 0, 0, 0);
62  // store result
63  TTI_SFPSTORE(0, 12, ADDR_MOD_7, 0);
64  sfpi::dst_reg++;
65  }
