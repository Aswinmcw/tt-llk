46 template <DstSync Dst, bool is_fp32_dest_acc_en = false>
47 inline void _llk_pack_dest_section_done_()
48 {
49 #ifdef PERF_DUMP
50  if constexpr (MATH_PACK_DECOUPLE)
51  {
52  return;
53  }
54 #endif
55 
56  constexpr bool clear_dest = (Dst != DstSync::SyncTile16);
57 
58  if constexpr (clear_dest)
59  {
60  TTI_STALLWAIT(p_stall::STALL_MATH, p_stall::PACK); // wait for pack to finish
61 
62  if constexpr (Dst == DstSync::SyncFull)
63  {
64  constexpr uint32_t CLEAR_MODE = is_fp32_dest_acc_en ? p_zeroacc::CLR_ALL_32B : p_zeroacc::CLR_ALL;
65  TT_ZEROACC(CLEAR_MODE, ADDR_MOD_1, 0);
66  }
67  else
68  {
69  static_assert((Dst == DstSync::SyncHalf) || (Dst == DstSync::SyncTile2));
70  constexpr uint32_t CLEAR_MODE = is_fp32_dest_acc_en ? p_zeroacc::CLR_HALF_32B : p_zeroacc::CLR_HALF;
71  TT_ZEROACC(CLEAR_MODE, ADDR_MOD_1, (dest_offset_id) % 2);
72  }
73  }
74 
75  // Note: we should have already stalled math in non-tile dest modes due to clearing
76  constexpr uint32_t WaitRes = (Dst == DstSync::SyncTile16) ? (p_stall::PACK) : (p_stall::NONE);
77 
78  // Tell math that it can write again
79  _llk_packer_set_math_semaphore_<WaitRes>();
80 
81  constexpr bool flip_dest = ((Dst == DstSync::SyncHalf) || (Dst == DstSync::SyncTile2));
82 
83  if constexpr (flip_dest)
84  {
85  flip_packer_dest_offset_id();
86  select_packer_dest_registers<Dst>();
87  }
88 }

221 TT_ALWAYS_INLINE void _llk_pack_relu_config_(const std::uint32_t config)
222 {
223  ReluType mode = (config & 0xf) == 0 ? ReluType::NO_RELU : ((config & 0xf) == 3 ? ReluType::MAX_THRESHOLD_RELU : ReluType::MIN_THRESHOLD_RELU);
224  uint32_t val = ((config >> 16) << STACC_RELU_ReluThreshold_SHAMT) | (((uint32_t)mode) << STACC_RELU_ApplyRelu_SHAMT);

226  TTI_SETDMAREG(0, val >> 16, 0, HI_16(p_gpr_pack::TMP0));
227  TTI_STALLWAIT(p_stall::STALL_CFG, p_stall::PACK);
228  TTI_WRCFG(p_gpr_pack::TMP0, p_cfg::WRCFG_32b, STACC_RELU_ApplyRelu_ADDR32);

238 template <bool untilize = false, ReduceDim dim>
239 inline void _llk_pack_reduce_mask_config_()
240 {
241  ckernel::packer::pck_edge_offset_u pack_edge_offset = {.val = 0};
242 
243  // We initialize PCK_EDGE_OFFSET_SEC0 mask to clear out all the datums in the row
244  pack_edge_offset.f.mask = 0x0;
245  uint32_t row_set_mapping_1 = 0;
246  uint32_t edge_offset_sec1_mask = 0;
247 
248  if constexpr (dim == ReduceDim::REDUCE_ROW)
249  {
250  // PCK_EDGE_OFFSET_SEC1 mask will clear out all the datums in the row except the first one
251  edge_offset_sec1_mask = 0x0001;
252  if constexpr (untilize)
253  {
254  pack_edge_offset.f.tile_row_set_select_pack0 = 1;
255  pack_edge_offset.f.tile_row_set_select_pack1 = 1;
256  pack_edge_offset.f.tile_row_set_select_pack2 = 1;
257  pack_edge_offset.f.tile_row_set_select_pack3 = 1;
258  row_set_mapping_1 = 0x11111111; // each packer packs 1x32 row
259  }
260  else
261  {
262  // Packer 0 and 2 will use TILE_ROW_SET_MAPPING_1, while packer 1 and 3 will keep using
263  // TILE_ROW_SET_MAPPING_0 configuration which is the default one
264  pack_edge_offset.f.tile_row_set_select_pack0 = 1;
265  pack_edge_offset.f.tile_row_set_select_pack2 = 1;
266 
267  // TILE_ROW_SET_MAPPING_1 configuration sets all rows to use PCK_EDGE_OFFSET_SEC1 mask
268  row_set_mapping_1 = 0x55555555; // each packer packs 1x16 row
269  }
270  }
271  else if constexpr (dim == ReduceDim::REDUCE_COL)
272  {
273  // PCK_EDGE_OFFSET_SEC1 mask will pass through all the datums in the row as they are
274  edge_offset_sec1_mask = 0xffff;
275 
276  // Packer 0 and 1 will use TILE_ROW_SET_MAPPING_1, while packer 2 and 3 will keep using
277  // TILE_ROW_SET_MAPPING_0 configuration which is the default one
278  pack_edge_offset.f.tile_row_set_select_pack0 = 1;
279  pack_edge_offset.f.tile_row_set_select_pack1 = 1;
280 
281  if constexpr (untilize)
282  {
283  row_set_mapping_1 = 0x00000005; // each packer packs 1x32 row
284  }
285  else
286  {
287  // TILE_ROW_SET_MAPPING_1 configuration sets only first row to use PCK_EDGE_OFFSET_SEC1 mask
288  row_set_mapping_1 = 0x00000001; // each packer packs 1x16 row
289  }
290  }
291  else if constexpr (dim == ReduceDim::REDUCE_SCALAR)
292  {
293  // PCK_EDGE_OFFSET_SEC1 mask will clear out all the datums in the row except the first one
294  edge_offset_sec1_mask = 0x0001;
295  // Packer 0 will use TILE_ROW_SET_MAPPING_1, while packers 1,2 and 3 will keep using
296  // TILE_ROW_SET_MAPPING_0 configuration which is the default one
297  pack_edge_offset.f.tile_row_set_select_pack0 = 1;
298 
299  // TILE_ROW_SET_MAPPING_1 configuration sets only first row to use PCK_EDGE_OFFSET_SEC1 mask
300  row_set_mapping_1 = 0x00000001;
301  }
302 
303  // Initialize TMP registers with values we need to write in CFG registers
304  TTI_SETDMAREG(0, LOWER_HALFWORD(pack_edge_offset.val), 0, LO_16(p_gpr_pack::TMP0));
305  TTI_SETDMAREG(0, UPPER_HALFWORD(pack_edge_offset.val), 0, HI_16(p_gpr_pack::TMP0));
306  TTI_SETDMAREG(0, LOWER_HALFWORD(edge_offset_sec1_mask), 0, LO_16(p_gpr_pack::TMP_LO));
307  TTI_SETDMAREG(0, LOWER_HALFWORD(row_set_mapping_1), 0, LO_16(p_gpr_pack::TMP1));
308  TTI_SETDMAREG(0, UPPER_HALFWORD(row_set_mapping_1), 0, HI_16(p_gpr_pack::TMP1));
309 
310  // Wait for packer to finish to avoid breaking its current configuration
311  TTI_STALLWAIT(p_stall::STALL_CFG, p_stall::PACK);
312 
313  // Configure packer
314  TTI_WRCFG(p_gpr_pack::TMP0, p_cfg::WRCFG_32b, PCK_EDGE_OFFSET_SEC0_mask_ADDR32);
315  TTI_WRCFG(p_gpr_pack::TMP_LO, p_cfg::WRCFG_32b, PCK_EDGE_OFFSET_SEC1_mask_ADDR32);
316  TTI_WRCFG(p_gpr_pack::TMP1, p_cfg::WRCFG_32b, TILE_ROW_SET_MAPPING_1_row_set_mapping_0_ADDR32);
317 
318  TTI_NOP;
319  TTI_NOP;
320 }
