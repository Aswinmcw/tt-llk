28 #endif
29 
30  if constexpr (BType == BroadcastType::COL)
31  {
32  static constexpr uint unpack_srcb_set_z = TT_OP_SETADCZW(0b010, 0, 0, 0, 2, 0b0001);
33  const uint32_t outerloop = num_faces < 4 ? 1 : 2;
34  const uint32_t innerloop = num_faces < 2 ? 1 : 2;
35  ckernel_template tmp(outerloop, innerloop, unpack_srca);
36  tmp.set_start_op(unpack_srcb);
37  if (narrow_tile)
38  {
39  tmp.set_end_op(unpack_srcb); // Read face 1

41  else

43  tmp.set_end_op(unpack_srcb_set_z);

46  }

50  static constexpr uint unpack_srcb_no_z_inc = TT_OP_UNPACR(SrcB, 0b0, 0, 0, 0, 1, 1, p_unpacr::RAREFYB_DISABLE, 0, 0, 0, 0, 1);

56  }
57  else if constexpr (BType == BroadcastType::SCALAR)
