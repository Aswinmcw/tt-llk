103 inline void dbg_thread_halt()
104 {
105  static_assert(
106  (thread_id == ThreadId::MathThreadId) || (thread_id == ThreadId::UnpackThreadId) || (thread_id == ThreadId::PackThreadId),
107  "Invalid thread id set in dbg_wait_for_thread_idle(...)");
108 
109  if constexpr (thread_id == ThreadId::UnpackThreadId)
110  {
111  // Wait for all instructions on the running thread to complete
112  tensix_sync();
113  // Notify math thread that unpack thread is idle
114  mailbox_write(ThreadId::MathThreadId, 1);
115  // Wait for math thread to complete debug dump
116  volatile uint32_t temp = mailbox_read(ThreadId::MathThreadId);
117  }
118  else if constexpr (thread_id == ThreadId::MathThreadId)
119  {
120  // Wait for all instructions on the running thread to complete
121  tensix_sync();
122  // Wait for unpack thread to complete
123  volatile uint32_t temp = mailbox_read(ThreadId::UnpackThreadId);
124  // Wait for previous packs to finish
125  while (semaphore_read(semaphore::MATH_PACK) > 0)
126  {
127  };
128  }

157 inline void dbg_get_array_row(const uint32_t array_id, const uint32_t row_addr, uint32_t *rd_data)
158 {
159  // Dest offset is added to row_addr to dump currently used half of the dest accumulator (SyncHalf dest mode)
160  std::uint32_t dest_offset = 0;
161  if (array_id == dbg_array_id::DEST)
162  {
163  dest_offset = (dest_offset_id == 1) ? DEST_REGISTER_HALF_SIZE : 0;
164  }
165 
166  if (array_id == dbg_array_id::SRCA)
167  {
168  // Save dest row
169  // WWhen SrcA array is selected we need to copy row from src register into dest to be able to dump data
170  // Dump from SrcA array is not supported
171  // Save dest row to SFPU register
172  // Move SrcA into dest row
173  // Dump dest row
174  // Restore dest row
175  addr_mod_t {
176  .srca = {.incr = 0, .clr = 0, .cr = 0},
177  .srcb = {.incr = 0, .clr = 0, .cr = 0},
178  .dest = {.incr = 0, .clr = 0, .cr = 0},
179  }
180  .set(ADDR_MOD_0);
181 
182  // Clear counters
183  TTI_SETRWC(p_setrwc::CLR_NONE, 0, 0, 0, 0, p_setrwc::SET_ABD_F);
184  TTI_SFPLOAD(p_sfpu::LREG3, 0, 0, 0); // Save dest addr 0 (even cols) to LREG_3
185  TTI_SFPLOAD(p_sfpu::LREG3, 0, 0, 2); // Save dest addr 0 (odd cols) to LREG_3
186 
187  TTI_STALLWAIT(p_stall::STALL_MATH, p_stall::SFPU1);
188 
189  // Move to the last used bank
190  TTI_CLEARDVALID(0b01, 0);
191 
192  // Copy single row from SrcA[row_addr] to dest location 0
193  TT_MOVDBGA2D(0, row_addr, 0, p_mova2d::MOV_1_ROW, 0);
194 
195  // Wait for TT instructions to complete
196  tensix_sync();
197  }
198  else if (array_id == dbg_array_id::SRCB)
199  {
200  addr_mod_t {
201  .srca = {.incr = 0, .clr = 0, .cr = 0},
202  .srcb = {.incr = 0, .clr = 0, .cr = 0},
203  .dest = {.incr = 0, .clr = 0, .cr = 0},
204  }
205  .set(ADDR_MOD_0);
206 
207  // Clear counters
208  TTI_SETRWC(p_setrwc::CLR_NONE, 0, 0, 0, 0, p_setrwc::SET_ABD_F);
209 
210  // Move to the last used bank
211  TTI_SETDVALID(0b10);
212  TTI_CLEARDVALID(0b10, 0);
213 
214  // Latch debug values
215  TTI_SETDVALID(0b10);
216  TTI_SHIFTXB(ADDR_MOD_0, 0, row_addr >> 1);
217  TTI_CLEARDVALID(0b10, 0);
218 
219  // Wait for TT instructions to complete
220  tensix_sync();
221 
222  /*
223  // Get last used bank id via debug bus
224  dbg_bus_cntl_u dbg_bus_cntl;
225  dbg_bus_cntl.val = 0;
226  dbg_bus_cntl.f.rd_sel = 6; // Read bits 111:96, 110 (write_math_id_reg)
227  dbg_bus_cntl.f.sig_sel = 0x4<<1;
228  dbg_bus_cntl.f.daisy_sel = dbg_daisy_id::INSTR_ISSUE_2;
229  dbg_bus_cntl.f.en = 1;
230  reg_write(RISCV_DEBUG_REG_DBG_BUS_CNTL_REG, dbg_bus_cntl.val);
231  wait (5); // Wait for value to get stable
232  srcb_bank_id = ((reg_read(RISCV_DEBUG_REG_DBG_RD_DATA)) & 0x4000) ? 0 : 1;
233 
234  // Disable debug bus
235  dbg_bus_cntl.val = 0;
236  reg_write(RISCV_DEBUG_REG_DBG_BUS_CNTL_REG, dbg_bus_cntl.val);
237  */
238  }
239 
240  // Get actual row address and array id used in hw
241  std::uint32_t hw_row_addr = (array_id == dbg_array_id::SRCA) ? 0 : ((array_id == dbg_array_id::DEST) ? dest_offset + row_addr : row_addr & 0x1);
242 
243  std::uint32_t hw_array_id = (array_id == dbg_array_id::SRCA) ? dbg_array_id::DEST : array_id;
244 
245  std::uint32_t hw_bank_id = 0;
246 
247  bool sel_datums_15_8 = hw_array_id != dbg_array_id::DEST;
248 
249  dbg_array_rd_en_u dbg_array_rd_en;
250  dbg_array_rd_en.val = 0;
251  dbg_array_rd_en.f.en = 0x1;
252  reg_write(RISCV_DEBUG_REG_DBG_ARRAY_RD_EN, dbg_array_rd_en.val);
253 
254  dbg_array_rd_cmd_u dbg_array_rd_cmd;
255  dbg_array_rd_cmd.val = 0;
256  dbg_array_rd_cmd.f.array_id = hw_array_id;
257  dbg_array_rd_cmd.f.bank_id = hw_bank_id;
258 
259  for (uint32_t i = 0; i < 8; i++)
260  {
261  dbg_array_rd_cmd.f.row_addr = sel_datums_15_8 ? (hw_row_addr | ((i >= 4) ? (1 << 6) : 0)) : hw_row_addr;
262  dbg_array_rd_cmd.f.row_32b_sel = i;
263  reg_write(RISCV_DEBUG_REG_DBG_ARRAY_RD_CMD, dbg_array_rd_cmd.val);
264  wait(5); // Wait for value to get stable
265  rd_data[i] = reg_read(RISCV_DEBUG_REG_DBG_ARRAY_RD_DATA);
266  }
267 
268  // Disable debug control
269  dbg_array_rd_cmd.val = 0;
270  reg_write(RISCV_DEBUG_REG_DBG_ARRAY_RD_CMD, dbg_array_rd_cmd.val);
271  dbg_array_rd_en.val = 0;
272  reg_write(RISCV_DEBUG_REG_DBG_ARRAY_RD_EN, dbg_array_rd_en.val);
273 
274  // Restore dest row
275  if (array_id == dbg_array_id::SRCA)
276  {
277  TTI_SFPSTORE(p_sfpu::LREG3, 0, 0, 0); // Restore dest addr 0 (even cols) from LREG_3
278  TTI_SFPSTORE(p_sfpu::LREG3, 0, 0, 2); // Restore dest addr 0 (odd cols) from LREG_3
279  // Move to the current bank
280  TTI_CLEARDVALID(1, 0);
281  }
282 }

284 inline std::uint32_t dbg_read_cfgreg(const uint32_t cfgreg_id, const uint32_t addr)
285 {
286  uint32_t hw_base_addr = 0;
287 
288  switch (cfgreg_id)
289  {
290  case dbg_cfgreg::HW_CFG_1:
291  hw_base_addr = dbg_cfgreg::HW_CFG_SIZE;
292  break;
293  case dbg_cfgreg::THREAD_0_CFG:
294  case dbg_cfgreg::THREAD_1_CFG:
295  case dbg_cfgreg::THREAD_2_CFG:
296  hw_base_addr = 2 * dbg_cfgreg::HW_CFG_SIZE + cfgreg_id * dbg_cfgreg::THREAD_CFG_SIZE;
297  break;
298  default:
299  break;
300  }
301 
302  uint32_t hw_addr = hw_base_addr + (addr & 0x7ff); // hw address is 4-byte aligned
303  reg_write(RISCV_DEBUG_REG_TENSIX_CREG_READ, hw_addr);
304 
305  wait(1);
306 
307  return reg_read(RISCV_DEBUG_REG_TENSIX_CREG_RDDATA);
308 }
