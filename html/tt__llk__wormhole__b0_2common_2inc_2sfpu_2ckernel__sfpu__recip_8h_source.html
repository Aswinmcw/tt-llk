14 template <int max_iter = 3>
15 sfpi_inline sfpi::vFloat _sfpu_reciprocal_(const sfpi::vFloat in)
16 {
17  // Force sign to 1 (make number negative)
18  sfpi::vFloat val = sfpi::setsgn(in, 1);
19 
20  val = setexp(val, 126); // Set exponent to 126 to make the number in 0.5-1
21  // Use 1.44 as first guess at x, ideal value would be 1.33.
22  // Grayskull has hardwired 1.44 and uses it to avoid a load.
23  // We use it here for consistency.
24  sfpi::vFloat vConstLn2Recip = sfpi::vConstFloatPrgm0;
25  sfpi::vFloat two = sfpi::vConstFloatPrgm1;
26  sfpi::vFloat result = vConstLn2Recip * (val * vConstLn2Recip + two);
27 
28  for (int s_iter = 0; s_iter < (max_iter - 1); s_iter++)
29  {
30  result = result * (val * result + two);
31  }
32 
33  sfpi::vInt orig_exp = exexp(in);
34  sfpi::vInt new_exp = exexp(result);
35 
36  // "Subtract" exponents, and re-bias.
37  // Execute: -1 - exp, then exp += 127
38  new_exp -= orig_exp;
39  new_exp += 126;
40 
41  v_if (new_exp < 0)
42  {
43  // If rebiased exponent is negative, we need to saturate at 0.
44  // This means the initial number was too big so reciprocal result should be 0
45  result = 0.0F;
46  new_exp = 0;
47  }
48  v_endif;
49 
50  // Set newly denormalized exponent to result exponent field
51  return setexp(result, new_exp);
52 }

54 template <bool APPROXIMATION_MODE, int ITERATIONS, bool is_fp32_dest_acc_en = true>
55 inline void _calculate_reciprocal_(const int iterations)
56 {
57 #pragma GCC unroll 8
58  for (int d = 0; d < iterations; d++)
59  {
60  sfpi::vFloat in = sfpi::dst_reg[0];
61  sfpi::vFloat out = _sfpu_reciprocal_<APPROXIMATION_MODE ? 2 : 3>(in);
62 
63  v_if (in < 0.0F)
64  {
65  // Invert sign on calculated value if CC=1 (number is negative)
66  out = -out;
67  }
68  v_endif;
69
