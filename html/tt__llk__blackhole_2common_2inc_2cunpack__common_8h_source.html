209 template <bool row_pool = false, bool is_fp32_dest_acc_en = false, bool fpu_srnd_en = false, bool pack_srnd_en = false, bool disable_src_zero_flag = false>
210 inline void configure_unpack_AB(
211  const uint unpA_src_format,
212  const uint unpB_src_format,
213  const uint unpA_dst_format,
214  const uint unpB_dst_format,
215  const uint unpA_face_r_dim = FACE_R_DIM,
216  const uint unpB_face_r_dim = FACE_R_DIM,
217  const bool transpose_xy_srca_en = false,
218  const uint unpA_num_faces = 4,
219  const uint unpB_num_faces = 4)
220 {
221  // Check that unpacker is done (all contexts freed up) before starting hw configuration
222  wait_for_idle();
223 
224  // Reset address counters
225  unpacker_addr_counter_init();
226 
227  const uint unpA_src_format_masked = (uint)unpA_src_format & 0x0F;
228  const uint unpB_src_format_masked = (uint)unpB_src_format & 0x0F;
229  const uint unpA_dst_format_masked = (uint)unpA_dst_format & 0x0F;
230  const uint unpB_dst_format_masked = (uint)unpB_dst_format & 0x0F;
231 
232  // Get pointer to registers for current state ID
233  volatile uint tt_reg_ptr *cfg = get_cfg_pointer();
234 
235  uint unpA_ch1_x_stride = (uint)(unpA_dst_format_masked & 0x3) == (uint)DataFormat::Float32 ? 4
236  : (uint)(unpA_dst_format_masked & 0x3) == (uint)DataFormat::Float16 ? 2
237  : 1;
238  uint unpB_ch1_x_stride = (uint)(unpB_dst_format_masked & 0x3) == (uint)DataFormat::Float32 ? 4
239  : (uint)(unpB_dst_format_masked & 0x3) == (uint)DataFormat::Float16 ? 2
240  : 1;
241  uint unpA_ch1_z_stride = FACE_C_DIM * FACE_R_DIM * unpA_ch1_x_stride;
242  uint unpB_ch1_z_stride = FACE_C_DIM * FACE_R_DIM * unpB_ch1_x_stride;
243  uint exp_width = ((uint)unpA_dst_format_masked >> 2) & 0x1; // 0=5-bit, 1=8-bit
244 
245  // Strides for incrementing ch1 address to srcA and srcB
246  cfg[UNP0_ADDR_CTRL_ZW_REG_1_Zstride_ADDR32] =
247  (0 << UNP0_ADDR_CTRL_ZW_REG_1_Wstride_SHAMT) |
248  (unpA_ch1_z_stride << UNP0_ADDR_CTRL_ZW_REG_1_Zstride_SHAMT); // Z and W(not used) stride for dest address (ch1)
249 
250  cfg[UNP1_ADDR_CTRL_ZW_REG_1_Zstride_ADDR32] =
251  (0 << UNP1_ADDR_CTRL_ZW_REG_1_Wstride_SHAMT) |
252  (unpB_ch1_z_stride << UNP1_ADDR_CTRL_ZW_REG_1_Zstride_SHAMT); // Z and W(not used) stride for dest address (ch1)
253 
254  // Math ALU_FORMAT_REG
255  t6_mutex_acquire(mutex::REG_RMW);
256  uint alu_src_format = (0x0 << ALU_FORMAT_SPEC_REG_SrcA_val_SHAMT);
257 
258  constexpr uint mask0 = (1 << (ALU_FORMAT_SPEC_REG_Dstacc_override_SHAMT + 1)) - 1;
259  cfg_reg_rmw_tensix<ALU_FORMAT_SPEC_REG_SrcA_val_ADDR32, ALU_FORMAT_SPEC_REG_SrcA_val_SHAMT, mask0>(alu_src_format);
260 
261  alu_config_u alu_payload = {.val = 0};
262 
263  uint32_t fp32_dest_acc_en = (is_fp32_dest_acc_en) ? (1) : (0);
264  uint32_t int8_math_enabled = ((uint)unpA_dst_format_masked == (uint)DataFormat::Int8) || ((uint)unpB_dst_format_masked == (uint)DataFormat::Int8) ||
265  ((uint)unpA_dst_format_masked == (uint)DataFormat::Int32) || ((uint)unpB_dst_format_masked == (uint)DataFormat::Int32);
266 
267  constexpr uint alu_format_mask = ALU_FORMAT_SPEC_REG0_SrcAUnsigned_MASK | ALU_FORMAT_SPEC_REG0_SrcBUnsigned_MASK;
268 
269  if ((uint)unpA_src_format == (uint)DataFormat::UInt8)
270  {
271  alu_payload.f.ALU_FORMAT_SPEC_REG0_SrcAUnsigned = 1;
272  }
273  if ((uint)unpB_src_format == (uint)DataFormat::UInt8)
274  {
275  alu_payload.f.ALU_FORMAT_SPEC_REG0_SrcBUnsigned = 1;
276  }
277 
278  // FP32 accumulation and SFPU to read dest as FP32
279  // NOTE: This assumes these config fields are adjacent and in same register!!
280  static_assert(ALU_ACC_CTRL_Fp32_enabled_ADDR32 == ALU_FORMAT_SPEC_REG0_SrcA_ADDR32);
281  static_assert(ALU_ACC_CTRL_Fp32_enabled_ADDR32 == ALU_ACC_CTRL_SFPU_Fp32_enabled_ADDR32);
282  constexpr uint alu_dest_format_mask = ALU_ACC_CTRL_SFPU_Fp32_enabled_MASK | ALU_ACC_CTRL_Fp32_enabled_MASK;
283  alu_payload.f.ALU_ACC_CTRL_Fp32_enabled = fp32_dest_acc_en;
284  alu_payload.f.ALU_ACC_CTRL_SFPU_Fp32_enabled = fp32_dest_acc_en;
285  constexpr uint alu_stoch_rnd_mask = ALU_ROUNDING_MODE_Fpu_srnd_en_MASK | ALU_ROUNDING_MODE_Gasket_srnd_en_MASK | ALU_ROUNDING_MODE_Packer_srnd_en_MASK;
286  alu_payload.f.ALU_ROUNDING_MODE_Fpu_srnd_en = fpu_srnd_en;
287  alu_payload.f.ALU_ROUNDING_MODE_Gasket_srnd_en = pack_srnd_en;
288  alu_payload.f.ALU_ROUNDING_MODE_Packer_srnd_en = pack_srnd_en;
289 
290  constexpr uint alu_mask = alu_format_mask | alu_dest_format_mask | alu_stoch_rnd_mask;
291 
292  cfg_reg_rmw_tensix<ALU_FORMAT_SPEC_REG0_SrcA_ADDR32, 0, alu_mask>(alu_payload.val);
293 
294  uint32_t src_zeroflags_disable = ((uint)unpA_dst_format == (uint)DataFormat::UInt16) || ((uint)unpB_dst_format == (uint)DataFormat::UInt16);
295  if constexpr (disable_src_zero_flag)
296  {
297  src_zeroflags_disable = true;
298  }
299  cfg_reg_rmw_tensix<ALU_ACC_CTRL_Zero_Flag_disabled_src_RMW>(src_zeroflags_disable);
300 
301  // Set FP8 E4M3 mode, bit is accessible by unpacker/packer
302  if ((unpA_src_format & 0x1F) == (uint)DataFormat::Fp8_e4m3)
303  {
304  cfg_reg_rmw_tensix<THCON_SEC0_REG1_Unp_LF8_4b_exp_RMW>(1);
305  }
306 
307  if ((unpB_src_format & 0x1F) == (uint)DataFormat::Fp8_e4m3)
308  {
309  cfg_reg_rmw_tensix<THCON_SEC1_REG1_Unp_LF8_4b_exp_RMW>(1);
310  }
311 
312  t6_mutex_release(mutex::REG_RMW);
313 
314  // Set tile descriptor
315  unpack_tile_descriptor_u tile_descriptor;
316  for (uint i = 0; i < TILE_DESC_SIZE; i++)
317  {
318  tile_descriptor.val[i] = 0;
319  }
320  tile_descriptor.f.in_data_format = (uint)unpA_src_format_masked;
321  tile_descriptor.f.uncompressed = 1; // Input tile is uncompressed
322  tile_descriptor.f.x_dim = 0; // Not used for unpA as value is overriden by per context x_dim set below. Used for unpB
323  tile_descriptor.f.y_dim = 1;
324  tile_descriptor.f.z_dim = unpA_num_faces;
325  // tile_descriptor.f.blobs_per_xy_plane = 0;
326  // tile_descriptor.f.blobs_y_start = 0;
327  for (uint i = 0; i < TILE_DESC_SIZE; i++)
328  {
329  cfg[THCON_SEC0_REG0_TileDescriptor_ADDR32 + i] = tile_descriptor.val[i];
330  }
331  tile_descriptor.f.in_data_format = row_pool ? (uint)DataFormat::Float32 : unpB_src_format_masked;
332  tile_descriptor.f.x_dim = unpB_face_r_dim * FACE_C_DIM;
333  tile_descriptor.f.z_dim = unpB_num_faces;
334  for (uint i = 0; i < TILE_DESC_SIZE; i++)
335  {
336  cfg[THCON_SEC1_REG0_TileDescriptor_ADDR32 + i] = tile_descriptor.val[i];
337  }
338 
339  // Set unpacker config
340  unpack_config_u config;
341  for (uint i = 0; i < CONFIG_SIZE; i++)
342  {
343  config.val[i] = 0;
344  }
345  config.f.out_data_format = unpA_dst_format_masked;
346  config.f.throttle_mode = 2;
347  config.f.context_count = 0;
348  config.f.haloize_mode = transpose_xy_srca_en ? 1 : 0;
349  // config.f.upsample_rate = 0;
350  // config.f.upsamle_and_interlave = 0;
351  // config.f.shift_amount = 0;
352  config.f.uncompress_cntx0_3 = 0xf;
353  config.f.uncompress_cntx4_7 = 0xf;
354  // config.f.limit_addr = 0; // Set dynamically
355  // config.f.fifo_size = 0; // Set dynamically
356  for (uint i = 0; i < CONFIG_SIZE; i++)
357  {
358  cfg[THCON_SEC0_REG2_Out_data_format_ADDR32 + i] = config.val[i];
359  }
360 
361  config.f.out_data_format = row_pool ? ((uint)DataFormat::Float16 | (exp_width << 2)) : unpB_dst_format_masked;
362  config.f.haloize_mode = 0;
363 
364  for (uint i = 0; i < CONFIG_SIZE; i++)
365  {
366  cfg[THCON_SEC1_REG2_Out_data_format_ADDR32 + i] = config.val[i];
367  }
368 
369  uint unpA_x_end = (unpA_face_r_dim == 0) ? 1 : (unpA_face_r_dim << 4) - 1;
370  TTI_SETADCXX(p_setadc::UNP_A, unpA_x_end, 0x0);
371  TTI_SETADCXX(p_setadc::UNP_B, (unpB_face_r_dim << 4) - 1, 0x0);
372 
373  // Program base address for all 2 sections (each section address is loaded to corresponding context)
374  // Load dummy data to unused location if face height is 0
375  const uint Dest_cntx0_address = unpA_face_r_dim == 0 ? 22 * 16 : 4 * 16;
376  const uint Dest_cntx1_address = unpA_face_r_dim == 0 ? 22 * 16 : 4 * 16;
377  cfg[THCON_SEC0_REG5_Dest_cntx0_address_ADDR32] = Dest_cntx0_address | (Dest_cntx1_address << 16);
378 
379  // Program unpacker0 per context x_dim (face size in l1)
380  // Overrides value set by tile descriptor when thread override bit is set in unpack instruction
381  const uint face_dim = unpA_face_r_dim * FACE_C_DIM;
382  cfg[THCON_SEC0_REG5_Tile_x_dim_cntx0_ADDR32] = face_dim | (face_dim << 16);
383 
384  constexpr uint face_dim_16x16 = FACE_R_DIM * FACE_C_DIM;
385  regfile[p_gpr_unpack::FACE_DIM_16x16] = (face_dim_16x16 / 1) | ((face_dim_16x16 / 1) << 16);
386  regfile[p_gpr_unpack::FACE_DIM_8x16] = (face_dim_16x16 / 2) | ((face_dim_16x16 / 2) << 16);
387  regfile[p_gpr_unpack::FACE_DIM_4x16] = (face_dim_16x16 / 4) | ((face_dim_16x16 / 4) << 16);
388  regfile[p_gpr_unpack::FACE_DIM_2x16] = (face_dim_16x16 / 8) | ((face_dim_16x16 / 8) << 16);
389  regfile[p_gpr_unpack::FACE_DIM_1x16] = (face_dim_16x16 / 16) | ((face_dim_16x16 / 16) << 16);
390  sync_regfile_write(p_gpr_unpack::FACE_DIM_1x16);
391 
392  TTI_SETC16(SRCA_SET_Base_ADDR32, 0x4);
393 
394  // Enable address counter for unpacker ch1/dst address
395  // final address is calculated as: Dest_cntx0/1_address + address_counter_ch1
396  // used for face by face unpacking of entire tile into srcA
397  cfg[UNP0_ADD_DEST_ADDR_CNTR_add_dest_addr_cntr_ADDR32] = 0x1 << UNP0_ADD_DEST_ADDR_CNTR_add_dest_addr_cntr_SHAMT;
398 
399  /*
400  // Workaround for HW bug (fp32 dest and movd2a/b is used with srcA/B configured with 5-bit exponent)
401  if (is_fp32_dest_acc_en && (exp_width == 0)) {
402  reg_write(RISCV_DEBUG_REG_DBG_FEATURE_DISABLE, 1<<11); // Set debug feature disable bit 11
403  // workaround for bug tenstorrent/budabackend#1372
404  }
405  */
406  // Workaround for HW bug (int32 dest and movd2a/b is used with srcA/B configured as int8)
407  if (int8_math_enabled || (fp32_dest_acc_en && ((uint)unpA_dst_format == (uint)DataFormat::UInt16)))
408  {
409  reg_write(RISCV_DEBUG_REG_DBG_FEATURE_DISABLE, 1 << 11); // Set debug feature disable bit 11
410  // workaround for bug tenstorrent/budabackend#1948
411  }
412 
413  // Clear context ID
414  reset_config_context();
415 }

449  // TTI_REG2FLOP(1,0,0,0,THCON_SEC0_REG5_Tile_x_dim_cntx0_ADDR32-THCON_CFGREG_BASE_ADDR32, p_gpr_unpack::FACE_DIM_1x16);

507 inline void set_dst_write_addr(const uint32_t &context_id, const uint32_t &unpack_dst_format)

510  TTI_SETC16(SRCA_SET_Base_ADDR32, 0x0); // Disable address bit swizzle
511  TTI_RDCFG(p_gpr_unpack::UNPACK_STRIDE, UNP0_ADDR_CTRL_ZW_REG_1_Zstride_ADDR32); // Save current stride

514  : 1;

516  TT_SETDMAREG(0, LOWER_HALFWORD(unpA_ch1_z_stride << UNP0_ADDR_CTRL_ZW_REG_1_Zstride_SHAMT), 0, LO_16(p_gpr_unpack::TMP_LO));
517  TTI_WRCFG(p_gpr_unpack::TMP_LO, p_cfg::WRCFG_32b, UNP0_ADDR_CTRL_ZW_REG_1_Zstride_ADDR32); // Set unpack stride
518  if (context_id == 0)

521  cfg_reg_rmw_tensix<THCON_SEC0_REG5_Dest_cntx0_address_RMW>(dst_byte_addr);
