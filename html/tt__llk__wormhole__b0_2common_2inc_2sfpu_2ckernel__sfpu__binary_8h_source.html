27 sfpi_inline sfpi::vFloat _calculate_sfpu_binary_power_(sfpi::vFloat base, sfpi::vFloat pow)
28 {
29  sfpi::vFloat original_base = base;
30 
31  // Check for integer power
32  sfpi::vInt pow_int = float_to_int16(pow, 0); // int16 should be plenty, since large powers will approach 0/Inf
33  sfpi::vFloat pow_rounded = int32_to_float(pow_int, 0);
34  v_if (pow_rounded == pow)
35  {
36  // if pow is integer, set base to positive
37  base = sfpi::setsgn(base, 0);
38  }
39  v_endif;
40 
41  // Normalize base to calculation range
42  sfpi::vFloat x = setexp(base, 127); // set exp to exp bias (put base in range of 1-2)
43 
44  // 3rd order polynomial approx - determined using rminimax over [1,2]
45  sfpi::vFloat series_result = x * (x * (x * 0x2.44734p-4f - 0xd.e712ap-4f) + 0x2.4f5388p+0f) - 0x1.952992p+0f;
46 
47  // Convert exponent to float
48  sfpi::vInt exp = exexp(base);
49  v_if (exp < 0)
50  {
51  exp = sfpi::setsgn(~exp + 1, 1);
52  }
53  v_endif;
54  sfpi::vFloat expf = int32_to_float(exp, 0);
55 
56  // De-normalize to original range
57  sfpi::vFloat vConstLn2 = 0.692871f;
58  sfpi::vFloat log_result = expf * vConstLn2 + series_result; // exp correction: ln(1+x) + exp*ln(2)
59 
60  // Base case when input is 0. ln(0) = -inf
61  v_if (base == 0.0f)
62  { // Reload for register pressure
63  log_result = -std::numeric_limits<float>::infinity();
64  }
65  v_endif;
66 
67  // Take exp(pow * log(base)) to produce base^pow
68  sfpi::vFloat val = pow * log_result;
69 
70  // Force sign to 0 (make number positive)
71  sfpi::vFloat result = _sfpu_exp_(sfpi::setsgn(val, 0));
72 
73  v_if (val < 0)
74  {
75  result = _sfpu_reciprocal_(result);
76  }
77  v_endif;
78 
79  // Check valid base range
80  v_if (original_base < 0.0f)
81  { // negative base
82  // Check for integer power
83  v_if (pow_rounded == pow)
84  {
85  // if pow is odd integer, set result to negative
86  v_if (pow_int & 0x1)
87  {
88  result = sfpi::setsgn(result, 1);
89  }
90  v_endif;
91  }
92  v_else
93  {
94  result = std::numeric_limits<float>::quiet_NaN();
95  }
96  v_endif;
97  }
98  v_endif;
99 
100  return result;
101 }

103 template <bool APPROXIMATION_MODE, BinaryOp BINOP, int ITERATIONS = 8>
104 inline void _calculate_sfpu_binary_(const uint dst_offset)
105 {
106  // SFPU microcode
107  for (int d = 0; d < ITERATIONS; d++)
108  {
109  constexpr uint dst_tile_size = 32;
110  sfpi::vFloat in0 = sfpi::dst_reg[0];
111  sfpi::vFloat in1 = sfpi::dst_reg[dst_offset * dst_tile_size];
112  sfpi::vFloat result = 0.0f;
113 
114  if constexpr (BINOP == BinaryOp::ADD)
115  {
116  result = in0 + in1;
117  }
118  else if constexpr (BINOP == BinaryOp::SUB)
119  {
120  result = in0 - in1;
121  }
122  else if constexpr (BINOP == BinaryOp::MUL)
123  {
124  result = in0 * in1;
125  }
126  else if constexpr (BINOP == BinaryOp::DIV)
127  {
128  v_if (in1 == 0)
129  {
130  v_if (in0 == 0)
131  {
132  result = std::numeric_limits<float>::quiet_NaN();
133  }
134  v_else
135  {
136  result = std::numeric_limits<float>::infinity();
137  result = sfpi::setsgn(result, in0);
138  }
139  v_endif;
140  }
141  v_elseif (in0 == in1)
142  {
143  result = sfpi::vConst1;
144  }
145  v_else
146  {
147  result = in0 * sfpi::setsgn(_sfpu_reciprocal_<4>(in1), in1);
148  }
149  v_endif;
150  }
151  else if constexpr (BINOP == BinaryOp::RSUB)
152  {
153  result = in1 - in0;
154  }
155  else if constexpr (BINOP == BinaryOp::POW)
156  {
157  result = _calculate_sfpu_binary_power_(in0, in1);
158  }
159 
160  sfpi::dst_reg[0] = result;
161  sfpi::dst_reg++;
162  }
163 }
