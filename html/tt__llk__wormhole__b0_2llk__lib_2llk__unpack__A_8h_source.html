215 inline void _llk_unpack_A_(
216  const std::uint32_t address, const bool transpose_of_faces = 0, const std::uint32_t unpack_src_format = 0, const std::uint32_t unpack_dst_format = 0)
217 {
218  // Clear z/w start counters
219  TTI_SETADCZW(0b011, 0, 0, 0, 0, 0b1111);
220 
221  // Program srcA and srcB base addresses
222  volatile uint tt_reg_ptr *cfg = get_cfg_pointer(); // get pointer to registers for current state ID
223 
224  // Wait for free context
225  wait_for_next_context(2);
226 
227  // Get tile address
228  if (0 == unp_cfg_context)
229  {
230  if constexpr ((BType == BroadcastType::NONE) && (!acc_to_dest))
231  {
232  cfg[THCON_SEC0_REG3_Base_address_ADDR32] = address;
233  }
234  else
235  {
236  if constexpr (binary_reuse_dest == EltwiseBinaryReuseDestType::DEST_TO_SRCB)
237  {
238  cfg[THCON_SEC0_REG3_Base_address_ADDR32] = address;
239  }
240  cfg[THCON_SEC1_REG3_Base_address_ADDR32] = address;
241  }
242  }
243  else
244  {
245  if constexpr ((BType == BroadcastType::NONE) && (!acc_to_dest))
246  {
247  cfg[THCON_SEC0_REG3_Base_cntx1_address_ADDR32] = address;
248  }
249  else
250  {
251  if constexpr (binary_reuse_dest == EltwiseBinaryReuseDestType::DEST_TO_SRCB)
252  {
253  cfg[THCON_SEC0_REG3_Base_cntx1_address_ADDR32] = address;
254  }
255  cfg[THCON_SEC1_REG3_Base_cntx1_address_ADDR32] = address;
256  }
257  }
258 
259  if constexpr (unpack_to_dest)
260  {
261  if (is_32bit_input(unpack_src_format, unpack_dst_format))
262  {
263  set_dst_write_addr(unp_cfg_context, unpack_dst_format);
264  wait_for_dest_available();
265  }
266  }
267 
268  // Trisc::SEMPOST for context acquire
269  semaphore_post(semaphore::UNPACK_SYNC);
270 
271  // Stall unpacker until pending CFG writes from Trisc have completed
272  TTI_STALLWAIT(p_stall::STALL_UNPACK, p_stall::TRISC_CFG);
273 
274  // Run MOP
275  ckernel::ckernel_template::run(instrn_buffer);
276 
277  // T6::SEMGET for context release
278  t6_semaphore_get(semaphore::UNPACK_SYNC);
279 
280  if (unpack_to_dest)
281  {
282  if (is_32bit_input(unpack_src_format, unpack_dst_format))
283  {
284  unpack_to_dest_tile_done(unp_cfg_context);
285  }
286  }
287 
288  // Switch unpacker config context
289  switch_config_context(unp_cfg_context);
290 
291 #ifdef PERF_DUMP
292  first_unpack_recorded = true;
293 #endif
294 }
