306 inline void wait(uint32_t cycles)
307 {
308  volatile uint tt_reg_ptr *clock_lo = reinterpret_cast<volatile uint tt_reg_ptr *>(RISCV_DEBUG_REG_WALL_CLOCK_L);
309  volatile uint tt_reg_ptr *clock_hi = reinterpret_cast<volatile uint tt_reg_ptr *>(RISCV_DEBUG_REG_WALL_CLOCK_H);
310  uint64_t wall_clock_timestamp = clock_lo[0] | ((uint64_t)clock_hi[0] << 32);
311  uint64_t wall_clock = 0;
312  do
313  {
314  wall_clock = clock_lo[0] | ((uint64_t)clock_hi[0] << 32);
315  } while (wall_clock < (wall_clock_timestamp + cycles));
316 }

343 inline void cfg_rmw(uint32_t cfg_addr32, uint32_t cfg_shamt, uint32_t cfg_mask, uint32_t val)
344 {
345  uint32_t wrdata = val;
346 
347  // Avoid multiplication of variables!
348  // const uint32_t addr = (cfg_state_id * CFG_STATE_SIZE * 4) + cfg_addr32;
349  const uint32_t addr = (cfg_state_id == 0) ? cfg_addr32 : (CFG_STATE_SIZE * 4) + cfg_addr32;
350 
351  // Declared here instead of globally to prevent direct access, which might ignore current state ID
352  volatile uint tt_reg_ptr *cfg_regs = reinterpret_cast<volatile uint tt_reg_ptr *>(TENSIX_CFG_BASE);
353  uint32_t cfg_data = cfg_regs[addr];
354 
355  // Shift and mask wrdata to properly align withn 32-bit DWORD
356  wrdata <<= cfg_shamt;
357  wrdata &= cfg_mask;
358 
359  // Zero-out relevant bits in cfg data
360  cfg_data &= ~cfg_mask;
361 
362  // Or new data bits
363  cfg_data |= wrdata;
364 
365  // Update cfg regs
366  cfg_regs[addr] = cfg_data;
367 }

375 template <uint CfgAddr32, uint Shamt, uint Mask>
376 inline void cfg_reg_rmw_tensix(uint32_t val)
377 {
378  uint32_t wrdata = val << Shamt;
379  uint8_t mask_b0 = Mask & 0xff;
380 
381  if (mask_b0 != 0)
382  {
383  uint8_t data_b0 = wrdata & 0xff;
384  TT_RMWCIB0(mask_b0, data_b0, CfgAddr32);
385  }
386  wrdata >>= 8;
387  uint8_t mask_b1 = (Mask >> 8) & 0xff;
388 
389  if (mask_b1 != 0)
390  {
391  uint8_t data_b1 = (wrdata) & 0xff;
392  TT_RMWCIB1(mask_b1, data_b1, CfgAddr32);
393  }
394 
395  wrdata >>= 8;

398  if (mask_b2 != 0)

400  uint8_t data_b2 = (wrdata) & 0xff;
