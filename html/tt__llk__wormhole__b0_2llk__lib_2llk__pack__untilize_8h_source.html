48 template <
49  std::uint32_t block_ct_dim,
50  std::uint32_t full_ct_dim = block_ct_dim,
51  bool diagonal = false,
52  bool narrow_row = false,
53  std::uint32_t row_num_datums = TILE_C_DIM>
54 inline void _llk_pack_untilize_mop_config_(const std::uint32_t face_r_dim = FACE_R_DIM, const std::uint32_t num_faces = 4)
55 {
56  const uint PACKCNT = diagonal ? (num_faces > 2 ? num_faces / 2 : num_faces) : ((face_r_dim < FACE_R_DIM) ? 1 : num_faces);
57  constexpr uint MEGAROW = 1;
58  constexpr uint ZERO_OUTPUT_FLAG = p_pacr::P_ZERO_OUTPUT_DISABLED;
59  constexpr uint MOP_INNER_LOOP = narrow_row ? (TILE_R_DIM / 4) : (diagonal ? FACE_R_DIM - 1 : 1);
60  constexpr uint MOP_OUTER_LOOP = narrow_row ? 1 : block_ct_dim;
61 
62  if constexpr (diagonal)
63  {
64  ckernel::ckernel_template tmp(
65  MOP_OUTER_LOOP,
66  MOP_INNER_LOOP,
67  TT_OP_INCADCXY(p_setadc::PAC, 0, 1, 0, 1),
68  TT_OP_PACR(ADDR_MOD_0, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 0));
69  tmp.set_start_op(TT_OP_PACR(ADDR_MOD_0, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 0));
70  tmp.set_last_inner_loop_instr(TT_OP_PACR(ADDR_MOD_1, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 0));
71  tmp.set_end_ops(TT_OP_SETADCXX(p_setadc::PAC, 1 - 1, 0x0), TT_OP_INCADCZW(p_setadc::PAC, 0, 0, 1, 0)); // w cnt points to the next tile
72  tmp.program(instrn_buffer);
73  }
74  else
75  {
76  if constexpr (narrow_row)
77  { // always
78  ckernel::ckernel_template tmp(MOP_OUTER_LOOP, MOP_INNER_LOOP, TT_OP_PACR(ADDR_MOD_0, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, 0, 0, 0));
79  tmp.program(instrn_buffer);
80  }
81  else
82  {
83  if (num_faces > 1)
84  {
85  // Inc ch0_y+=1 (addr_mod_0 will increment by 15)
86  ckernel::ckernel_template tmp(MOP_OUTER_LOOP, MOP_INNER_LOOP, TT_OP_INCADCXY(p_setadc::PAC, 0, 0, 1, 0));
87  tmp.set_start_op(TT_OP_PACR(ADDR_MOD_0, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 0));
88  tmp.set_end_ops(
89  TT_OP_PACR(ADDR_MOD_1, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 0),
90  TT_OP_INCADCZW(p_setadc::PAC, 0, 0, 1, 0)); // w cnt points to the next tile
91  tmp.program(instrn_buffer);
92  }
93  else
94  {
95  ckernel::ckernel_template tmp(MOP_OUTER_LOOP, MOP_INNER_LOOP, TT_OP_PACR(ADDR_MOD_1, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 0));
96  tmp.set_end_op(TT_OP_INCADCZW(p_setadc::PAC, 0, 0, 1, 0)); // w cnt points to the next tile
97  tmp.program(instrn_buffer);
98  }
99  }
100 
101  if (block_ct_dim != full_ct_dim)
102  {
103  const std::uint32_t replay_buf_len = 10;
104  TT_REPLAY(ckernel::packer::replay_buf_offset, replay_buf_len, 0, 1);
105  TTI_PACR(ADDR_MOD_2, 0, 0xf, 0, 0, 1, 1); // close block
106  // update l1 address
107  TTI_ADDDMAREG(0, p_gpr_pack::OUTPUT_ADDR, p_gpr_pack::OUTPUT_ADDR, p_gpr_pack::OUTPUT_ADDR_OFFSET);
108  TTI_ADDDMAREG(0, p_gpr_pack::OUTPUT_ADDR + 1, p_gpr_pack::OUTPUT_ADDR + 1, p_gpr_pack::OUTPUT_ADDR_OFFSET);
109  TTI_ADDDMAREG(0, p_gpr_pack::OUTPUT_ADDR + 2, p_gpr_pack::OUTPUT_ADDR + 2, p_gpr_pack::OUTPUT_ADDR_OFFSET);
110  TTI_ADDDMAREG(0, p_gpr_pack::OUTPUT_ADDR + 3, p_gpr_pack::OUTPUT_ADDR + 3, p_gpr_pack::OUTPUT_ADDR_OFFSET);
111  TTI_REG2FLOP(1, 0, 0, 0, THCON_SEC0_REG1_L1_Dest_addr_ADDR32 - THCON_CFGREG_BASE_ADDR32, p_gpr_pack::OUTPUT_ADDR);
112  TTI_REG2FLOP(1, 0, 0, 0, THCON_SEC0_REG8_L1_Dest_addr_ADDR32 - THCON_CFGREG_BASE_ADDR32, p_gpr_pack::OUTPUT_ADDR + 1);
113  TTI_REG2FLOP(1, 0, 0, 0, THCON_SEC1_REG1_L1_Dest_addr_ADDR32 - THCON_CFGREG_BASE_ADDR32, p_gpr_pack::OUTPUT_ADDR + 2);
114  TTI_REG2FLOP(1, 0, 0, 0, THCON_SEC1_REG8_L1_Dest_addr_ADDR32 - THCON_CFGREG_BASE_ADDR32, p_gpr_pack::OUTPUT_ADDR + 3);
115  TTI_NOP;
116  }
117  }
118 }

139 template <

145 inline void _llk_pack_untilize_(
146  const std::uint32_t address,
147  const std::uint32_t pack_dst_format,
148  const std::uint32_t face_r_dim = FACE_R_DIM,
149  const std::uint32_t num_faces = 4 /*not used*/,
150  const std::uint32_t tile_dst_offset = 0)
151 {
152  program_packer_untilized_destination<block_ct_dim, full_ct_dim, diagonal, row_num_datums>(address, pack_dst_format);
153 
154  if constexpr (narrow_row)
155  {
156  ckernel::ckernel_template::run(instrn_buffer);
157  }
158  else
159  {
160  const std::uint32_t num_rows = (face_r_dim < FACE_R_DIM) ? face_r_dim : TILE_R_DIM / 4;
161 
162  for (std::uint32_t row = 0; row < num_rows; row++)
163  {
164  TT_SETADC(p_setadc::PAC, p_setadc::CH_0, p_setadc::SET_W, tile_dst_offset); // Clear tile counter
165  ckernel::ckernel_template::run(instrn_buffer);
166  TTI_ADDRCRXY(p_setadc::PAC, 0, 0, 1, 0, 0b0010); // Read new row in the tile
167  if constexpr (block_ct_dim != full_ct_dim)
168  {
169  TTI_REPLAY(ckernel::packer::replay_buf_offset, 10, 0, 0); // update row address
170  }
171  }
172  }
173 
174  if constexpr (block_ct_dim == full_ct_dim)
175  {
176  TTI_PACR(ADDR_MOD_2, 0, 0xf, 0, 0, 1, 1); // close block
177  }
178 }
