16 sfpi_inline void _calculate_log_body_(const uint log_base_scale_factor)
17 {
19  // Load From dest + "normalize to calculation range"
21  sfpi::vFloat in = sfpi::dst_reg[0];
22  sfpi::vFloat x = setexp(in, 127); // set exp to exp bias (put in range of 1-2)
23 
24  // XXXXXX ask Namal? if we can derive the coefficients below to higher precision
26  // Calculate Cheby Approximation using Horner Form Multiplication: 3rd Order
27  // x* ( x* (A*x + B) + C) + D
28  // A :0.1058, B: -0.3942, C: 0.9813, D: 0.006
29  // Run above on (x-1) so x is in ln(x+1), plug (x-1 into equation above to
30  // save the subtract and get A',B',C',D'):
31  // A' = A
32  // B' = -3A + B
33  // C' = 3a -2B + C
34  // D' = -A + B - C + D
35  // A':0.1058, B':-0.7116, C':2.0871, D':-1.4753
37  sfpi::vFloat a = sfpi::vConstFloatPrgm1;
38  sfpi::vFloat b = sfpi::vConstFloatPrgm2;
39  // XXXXX try variants of the below: B'=.7122, C'=2.0869
40  sfpi::vFloat series_result = x * (x * (x * a + b) + 2.0871) + -1.4753f;
41 
43  // Convert exponent to float
45  sfpi::vInt exp = exexp(in);
46  v_if (exp < 0)
47  {
48  exp = sfpi::setsgn(~exp + 1, 1);
49  }
50  v_endif;
51 
52  sfpi::vFloat expf = int32_to_float(exp, 0);
53  sfpi::vFloat vConstLn2 = sfpi::vConstFloatPrgm0;
54  sfpi::vFloat result = expf * vConstLn2 + series_result; // exp correction: ln(1+x) + exp*ln(2)
55 
56  if constexpr (HAS_BASE_SCALING)
57  {
58  result *= sfpi::s2vFloat16a(log_base_scale_factor);
59  }
60 
62  // Base case when input is 0. ln(0) = -inf
64  v_if (in == 0.0F)
65  { // Reload for register pressure
66  result = -std::numeric_limits<float>::infinity();
67  }
68  v_endif;
69 
70  sfpi::dst_reg[0] = result;
71 }
