19 template <bool untilize_en = false, bool row_pool = false>
20 inline void _llk_math_hw_configure_(const std::uint32_t srca_data_format, const std::uint32_t srcb_data_format)
21 {
22  TTI_STALLWAIT(p_stall::STALL_CFG, p_stall::MATH | p_stall::WAIT_SFPU);
23  uint exp_width = ((uint)srca_data_format >> 2) & 0x1; // 0=5-bit, 1=8-bit
24  uint int8_math_enabled = ((uint)(srca_data_format & 0xF) == (uint)DataFormat::Int8) || ((uint)(srcb_data_format & 0xF) == (uint)DataFormat::Int8) ||
25  ((uint)srca_data_format == (uint)DataFormat::Int32) || ((uint)srcb_data_format == (uint)DataFormat::Int32);
26  uint srcb_format = (row_pool ? ((uint)DataFormat::Float16 | (exp_width << 2)) : srcb_data_format);
27  uint config_data = (srca_data_format << ALU_FORMAT_SPEC_REG0_SrcA_SHAMT) | (srcb_format << ALU_FORMAT_SPEC_REG1_SrcB_SHAMT) |
28  (int8_math_enabled << ALU_ACC_CTRL_INT8_math_enabled_SHAMT);
29  constexpr uint config_mask = ALU_FORMAT_SPEC_REG0_SrcA_MASK | ALU_FORMAT_SPEC_REG1_SrcB_MASK | ALU_ACC_CTRL_INT8_math_enabled_MASK;
30  cfg_reg_rmw_tensix<ALU_FORMAT_SPEC_REG0_SrcA_ADDR32, 0, config_mask>(config_data);
31 }

48 template <DstSync Dst, bool is_fp32_dest_acc_en = false>
49 inline void _llk_math_dest_section_done_()
50 {
51 #ifdef PERF_DUMP
52  if constexpr (MATH_PACK_DECOUPLE)
53  {
54  return;
55  }
56 #endif
57 
58  constexpr uint32_t DEST_NUM_TILES_SHIFT = is_fp32_dest_acc_en ? (1) : (0);
59  constexpr uint32_t DEST_NUM_TILES = DEST_NUM_TILES_FP16 >> DEST_NUM_TILES_SHIFT;
60 
61  set_math_semaphores();

64  math_sync_tile_dst_index = 0;

66  }
67  else if constexpr (Dst == DstSync::SyncTile16)

74 template <DstSync Dst, bool is_fp32_dest_acc_en = false>
75 inline void _llk_math_pack_sync_init_()
76 {
77 #ifdef PERF_DUMP
78  if constexpr (MATH_PACK_DECOUPLE)
79  {
80  return;
81  }
82 #endif
83  tensix_sync();
84  while (semaphore_read(semaphore::MATH_PACK) > 0)
85  {
86  }; // Wait for previous packs to finish before claiming all dest
87  if constexpr (Dst == DstSync::SyncFull)
88  {
89  TTI_SEMINIT(1, 0, p_stall::SEMAPHORE_1);
90  reset_dest_offset_id();
91  set_dest_section_base<StartZero>();
92  }
93  else if constexpr (Dst == DstSync::SyncHalf)
94  {
95  TTI_SEMINIT(2, 0, p_stall::SEMAPHORE_1);
96  reset_dest_offset_id();
97  set_dest_section_base<StartZero>();
98  }
99  else if constexpr (Dst == DstSync::SyncTile2)
100  {
101  TTI_SEMINIT(2, 0, p_stall::SEMAPHORE_1);
102  reset_dest_offset_id();
103  set_dest_section_base<StartZero>();
104  math_sync_tile_dst_index = 0;
105  }
106  else
107  {
108  static_assert(Dst == DstSync::SyncTile16);
109 
110  constexpr uint32_t DEST_NUM_TILES_SHIFT = is_fp32_dest_acc_en ? (1) : (0);
111  constexpr uint32_t DEST_NUM_TILES = DEST_NUM_TILES_FP16 >> DEST_NUM_TILES_SHIFT;
112  constexpr uint32_t SEM_INIT_MAX = (DEST_NUM_TILES < 15) ? DEST_NUM_TILES : 15;
113 
114  TTI_SEMINIT(SEM_INIT_MAX, 0, p_stall::SEMAPHORE_1);
115  reset_dest_offset_id();
116  set_dest_section_base<StartZero>();
117  math_sync_tile_dst_index = 0;
118  }
119 }
