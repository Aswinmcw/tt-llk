55 inline void _llk_pack_mop_config_(
56  const std::uint32_t pack_dst_format,
57  const std::uint32_t face_r_dim = FACE_R_DIM,
58  const std::uint32_t num_faces = 4,
59  const bool partial_face = false,
60  const bool narrow_tile = false)
61 {
62  static_assert(FaceLayout == DstTileFaceLayout::RowMajor, "FaceLayout must be RowMajor");
63 
64  const uint PACKCNT = (partial_face && IS_BFP_FORMAT(pack_dst_format)) ? 1 : num_faces;
65  constexpr uint MEGAROW = 1;
66  constexpr uint ZERO_OUTPUT_FLAG = zero_output ? p_pacr::P_ZERO_OUTPUT_ENABLED : p_pacr::P_ZERO_OUTPUT_DISABLED;
67  constexpr uint MOP_INNER_LOOP = 1;
68 
69  if constexpr (!untilize)
70  {
71  constexpr uint MOP_OUTER_LOOP = 1;
72 
73  ckernel::ckernel_template tmp(MOP_OUTER_LOOP, MOP_INNER_LOOP, TT_OP_PACR(ADDR_MOD_1, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 1));
74 
75  if (partial_face && IS_BFP_FORMAT(pack_dst_format))
76  {
77  tmp.set_start_op(TT_OP_PACR(ADDR_MOD_0, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 0)); // Don't close the tile, point to the next face
78  tmp.set_loop_op0(TT_OP_INCADCXY(p_setadc::PAC, 0, 0, 1, 0)); // Inc ch0_y+=1 (addr_mod_0 will increment by 15)
79  tmp.set_loop_op1(TT_OP_PACR(ADDR_MOD_1, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 1)); // Close the tile
80  }
81  // Write header to l1
82  if constexpr (write_tile_header)
83  {
84  tmp.set_end_op(TT_OP_STOREIND(1, 0, p_ind::LD_16B, LO_16(0), p_ind::INC_NONE, p_gpr_pack::TILE_HEADER, p_gpr_pack::OUTPUT_ADDR));
85  }
86 
87  tmp.program(instrn_buffer);
88  }
89  else
90  {
91  const uint MOP_OUTER_LOOP = ((face_r_dim == 1) || narrow_tile) ? 1 : (face_r_dim >> 1);
92 
93  if ((face_r_dim == 1) || narrow_tile)
94  {
95  ckernel::ckernel_template tmp(MOP_OUTER_LOOP, MOP_INNER_LOOP, TT_OP_PACR(ADDR_MOD_0, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 1));
96  tmp.program(instrn_buffer);
97  }
98  else
99  {
100  // Inc ch0_y+=1 (addr_mod_0 will increment by 15)
101  ckernel::ckernel_template tmp(MOP_OUTER_LOOP, MOP_INNER_LOOP, TT_OP_INCADCXY(p_setadc::PAC, 0, 0, 1, 0));
102  tmp.set_start_op(TT_OP_PACR(ADDR_MOD_0, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 0));
103  tmp.set_end_op(TT_OP_PACR(ADDR_MOD_1, ZERO_OUTPUT_FLAG, PACK_SEL(PACKCNT), 0, MEGAROW, 0, 0));
104  tmp.program(instrn_buffer);
105  }
106  }

233 template <DstSync Dst, bool untilize = false, bool is_fp32_dest_acc_en = false>
234 inline void _llk_pack_(const std::uint32_t tile_index, const std::uint32_t address)
235 {
236  constexpr uint32_t DEST_NUM_TILES_SHIFT = is_fp32_dest_acc_en ? (1) : (0);
237  constexpr uint32_t DEST_NUM_TILES = DEST_NUM_TILES_FP16 >> DEST_NUM_TILES_SHIFT;
238 
239  if constexpr (Dst == DstSync::SyncTile16)
240  {
241  // W-counter points to the next tile in dest
242  TT_SETADC(p_setadc::PAC, p_setadc::CH_0, p_setadc::SET_W, pack_sync_tile_dst_ptr);
243  pack_sync_tile_dst_ptr += 1;
244  pack_sync_tile_dst_ptr = pack_sync_tile_dst_ptr & (DEST_NUM_TILES - 1);
245  }
246  else if constexpr (Dst == DstSync::SyncTile2)
247  {
248  TT_SETADC(p_setadc::PAC, p_setadc::CH_0, p_setadc::SET_W, pack_sync_tile_dst_ptr);
249  pack_sync_tile_dst_ptr = 0;
250  }
251  else
252  {
253  TT_SETADC(p_setadc::PAC, p_setadc::CH_0, p_setadc::SET_W, tile_index);
254  }
255 
256  program_packer_destination(address);
257 
258  mop_run(1, 1);
259 
260  if constexpr (untilize)
261  {
262  TTI_PACR(ADDR_MOD_2, 0, 0xf, 0, 0, 1, 1); // close tile
263  }
264 }
