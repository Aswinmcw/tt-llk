316 inline void set_packer_l1_offset(const uint pack_dst_format, const uint face_r_dim = FACE_R_DIM)
317 {
318  const uint face_dim = face_r_dim * FACE_C_DIM;
319 
320  uint32_t l1_offset_1 = IS_BFP_FORMAT(pack_dst_format) ? 1
321  : (((uint8_t)(pack_dst_format & 0x3) == (uint8_t)DataFormat::Float32) ? (face_dim / 16) * 4
322  : ((uint8_t)(pack_dst_format & 0x3) == (uint8_t)DataFormat::Float16) ? (face_dim / 16) * 2
323  : (face_dim / 16));
324  uint32_t l1_offset_2 = 2 * l1_offset_1;
325  uint32_t l1_offset_3 = 3 * l1_offset_1;
326 
327  // HW automatically offsets packers base address by tile header size
328  // with new L1 addressing mode, the effective address for pack1/2/3
329  // will be pack[i] += pack[0], which leads to double counting of tile header
330  // subtract by this amount when programming the offset
331  constexpr uint32_t PACK_TILE_HEADER_OFFSET = 1; // in 16B
332  l1_offset_1 -= PACK_TILE_HEADER_OFFSET;
333  l1_offset_2 -= PACK_TILE_HEADER_OFFSET;
334  l1_offset_3 -= PACK_TILE_HEADER_OFFSET;
335  TT_SETDMAREG(0, LOWER_HALFWORD(l1_offset_1), 0, LO_16(p_gpr_pack::TMP_LO));
336  TTI_REG2FLOP(2, 0, 0, 0, THCON_SEC0_REG8_L1_Dest_addr_ADDR32 - THCON_CFGREG_BASE_ADDR32, p_gpr_pack::TMP_LO);
337  TT_SETDMAREG(0, LOWER_HALFWORD(l1_offset_2), 0, LO_16(p_gpr_pack::TMP_LO));

407  }

418 
419  TT_SETDMAREG(0, LOWER_HALFWORD(tile_size), 0, LO_16(p_gpr_pack::TILE_HEADER));
420
