17 template <bool APPROXIMATION_MODE>
18 inline sfpi::vFloat _calculate_gelu_core_(sfpi::vFloat in)
19 {
20  // SFPU microcode:
21  // result = (APPROX_MODE == 1)
22  // ? (1 + erf(x/sqrt(2)))
23  // : (1 + tanh( sqrt(2/pi) * (x + 0.044715*x^3) )
24  sfpi::vFloat result;
25  if constexpr (APPROXIMATION_MODE)
26  {
27  result = in;
28  }
29  else
30  {
31  // f = (0.044715*x^3 + x)
32  result = (in * in) * (in * sfpi::s2vFloat16b(0.044715f)) + in;
33  result *= sfpi::s2vFloat16b(0.79788f);
34  }
35 
36  return result;
37 }

87 template <bool APPROXIMATION_MODE, int ITERATIONS>
88 inline void _calculate_gelu_derivative_(const int iterations)
89 {
90  if constexpr (APPROXIMATION_MODE)
91  {
92  constexpr int lut_mode = 1; // SFPLUTFP32_MOD0_FP16_6ENTRY_TABLE1
93 
94  sfpi::vUInt l0 = sfpi::l_reg[sfpi::LRegs::LReg0];
95  sfpi::vUInt l1 = sfpi::l_reg[sfpi::LRegs::LReg1];
96  sfpi::vUInt l2 = sfpi::l_reg[sfpi::LRegs::LReg2];
97  sfpi::vUInt l4 = sfpi::l_reg[sfpi::LRegs::LReg4];
98  sfpi::vUInt l5 = sfpi::l_reg[sfpi::LRegs::LReg5];
99  sfpi::vUInt l6 = sfpi::l_reg[sfpi::LRegs::LReg6];
100 
101 // SFPU microcode:
102 #pragma GCC unroll 0
103  for (int d = 0; d < iterations; d++)
104  {
105  sfpi::vFloat val = sfpi::dst_reg[0];
106  val = lut2(val, l0, l1, l2, l4, l5, l6, lut_mode);
107  v_if (val < 0.0F)
108  {
109  val = val + 1.0f;
110  }
111  v_endif;
112  sfpi::dst_reg[0] = val;
113  sfpi::dst_reg++;
114  }
115 
116  sfpi::l_reg[sfpi::LRegs::LReg0] = l0;
117  sfpi::l_reg[sfpi::LRegs::LReg1] = l1;
118  sfpi::l_reg[sfpi::LRegs::LReg2] = l2;
119  sfpi::l_reg[sfpi::LRegs::LReg4] = l4;
120  sfpi::l_reg[sfpi::LRegs::LReg5] = l5;
121  sfpi::l_reg[sfpi::LRegs::LReg6] = l6;
122  }
123  else
124  {
125  constexpr uint imm2 = 0xFF10;
126 
127  sfpi::vUInt l0 = sfpi::l_reg[sfpi::LRegs::LReg0];
128  sfpi::vUInt l1 = sfpi::l_reg[sfpi::LRegs::LReg1];
129 
130 // SFPU microcode:
131 #pragma GCC unroll 0
132  for (int d = 0; d < iterations; d++)
133  {
134  sfpi::vFloat in = sfpi::dst_reg[0];
135  sfpi::vFloat neg_half_sq_in = in * in * -0.5f;
136 
137  // exp = e^(val)
138  sfpi::vFloat exp = _calculate_exponential_body_<false>(neg_half_sq_in);
139 
140  // exp = exp * 1/sqrt(2*pi)
141  sfpi::vFloat partial = exp * in * sfpi::s2vFloat16b(0.3989423F);
142 
143  sfpi::vFloat result = _calculate_gelu_core_<true>(in);
144 
145  result = lut(result, l0, l1, imm2);
146 
147  sfpi::dst_reg[0] = partial + result + 0.5f;
148  sfpi::dst_reg++;
149  }
150 
151  sfpi::l_reg[sfpi::LRegs::LReg0] = l0;
152  sfpi::l_reg[sfpi::LRegs::LReg1] = l1;
153  }
154 }
