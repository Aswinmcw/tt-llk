18 {
19  // Operand A is input1 (int32)
20  // Operand B is input2 (int32)
21  // Output is int32
22 
23  // Modifies LOAD/STORE to do INT32 sign-magnitude to 2's complement conversion, however
24  // in Blackhole this has no effect and format remains in original format.
25  constexpr auto INSTR_MOD_LOAD_STORE = InstrModLoadStore::INT32_2S_COMP;
26 
27  // If LOAD/STORE have the value in INT sign-magnitude format and SFPU needs it as 2's complement.
28  constexpr auto INSTR_MOD_CAST = InstrModCast::INT_SIGN_MAGN_TO_INT32_2S_COMP;
29 
30 #pragma GCC unroll 8
31  for (int d = 0; d < ITERATIONS; d++)
32  {
33  // operand B - int32
34  TT_SFPLOAD(0 /*lreg*/, INSTR_MOD_LOAD_STORE, ADDR_MOD_7, dst_offset * 64 /*dest_reg_addr */);
35  if constexpr (SIGN_MAGNITUDE_FORMAT)
36  {
37  TTI_SFPCAST(0 /*lreg*/, 2 /*ldest*/, INSTR_MOD_CAST);
38  // Required after cast due to a bug in Blackhole RTL.
39  TTI_SFPSETSGN(0 /* imm */, 2 /*lreg_c*/, 0 /*ldest*/, 0 /*imod*/);
40  }
41 
42  // operand A - int32
43  TTI_SFPLOAD(1 /*lreg*/, INSTR_MOD_LOAD_STORE, ADDR_MOD_7, 0);
44  if constexpr (SIGN_MAGNITUDE_FORMAT)
45  {
46  TTI_SFPCAST(1 /*lreg*/, 2 /*ldest*/, INSTR_MOD_CAST);
47  // Required after cast due to a bug in Blackhole RTL.
48  TTI_SFPSETSGN(0 /* imm */, 2 /*lreg_c*/, 1 /*ldest*/, 0 /*imod*/);
49  }
50 
51  // Set instruction modifier to 6 to get B's 2's complement
52  TTI_SFPIADD(0 /*imm*/, 1 /*lreg_c*/, 0 /*lreg_dest*/, 6 /*imod*/);
53 
54  // LREG_0 -> dest as int32
55  if constexpr (SIGN_MAGNITUDE_FORMAT)
56  {
57  TTI_SFPCAST(0 /*lreg*/, 1 /*ldest*/, INSTR_MOD_CAST);
58  // Required after cast due to a bug in Blackhole RTL.
59  TTI_SFPSETSGN(0 /* imm */, 1 /*lreg_c*/, 0 /*ldest*/, 0 /*imod*/);
60  }
61  TTI_SFPSTORE(0, INSTR_MOD_LOAD_STORE, ADDR_MOD_7, 0);
