132 template <DataCopyType type, BroadcastType bcast_type = BroadcastType::NONE, bool is_fp32_dest_acc_en = false, bool is_int_fpu_en = false>
133 inline void eltwise_unary_configure_mop(uint rows_per_inst, uint total_rows, const uint num_faces, const uint dst_format)
134 {
135  // always move 32x32 tile, packed as 16x16x4
136 
137  if constexpr (type == A2D)
138  {
139  uint addr_mod = (rows_per_inst == p_mova2d::MOV_1_ROW) ? ADDR_MOD_0 : ADDR_MOD_2;
140  uint innerloop = (rows_per_inst == p_mova2d::MOV_1_ROW) ? total_rows : (total_rows >> 3);
141  uint outerloop = num_faces;
142 
143  if (((is_fp32_dest_acc_en || is_int_fpu_en) && !(dst_format == (uint)DataFormat::UInt16)) || (dst_format == (uint)DataFormat::UInt8))
144  {
145  // use elwadd to handle unpacking data into src A as fp16, but dest is in fp32 mode OR to handle uint8 datums
146  ckernel_template tmp(outerloop, innerloop, TT_OP_ELWADD(0, 0, p_elwise::SRCB_NO_BCAST, ADDR_MOD_2, 0));
147  tmp.set_end_op(TT_OP_SETRWC(p_setrwc::CLR_AB, 0, 0, 0, 0, p_setrwc::SET_AB));
148  tmp.program(instrn_buffer);
149  }
150  else
151  {
152  ckernel_template tmp(outerloop, innerloop, TT_OP_MOVA2D(0, 0, ADDR_MOD_2, p_mova2d::MOV_8_ROWS, 0));
153  tmp.set_end_op(TT_OP_SETRWC(p_setrwc::CLR_AB, 0, 0, 0, 0, p_setrwc::SET_AB));
154  tmp.program(instrn_buffer);
155  }
156  }
157  else if constexpr (type == B2D)
158  {
159  uint addr_mod = (rows_per_inst == p_movb2d::MOV_1_ROW) ? ADDR_MOD_0 : ADDR_MOD_2;
160  uint innerloop = (rows_per_inst == p_movb2d::MOV_1_ROW) ? total_rows : (total_rows >> 2);
161  uint outerloop = 4;
162  auto broadcast_type = p_movb2d::MOV_1_ROW; // No broadcast;
163 
164  if constexpr (bcast_type == BroadcastType::COL)
165  {
166  innerloop = 16 >> 3; // elwadd produces 8 rows per op
167  // The mop only runs for 2 outer loops and mop is called twice for col broadcast
168  outerloop = 2;
169  // broadcast_type = p_movb2d::MOV_8_ROW_BRCST_D0_BRCST;
170  // MOVB2D with column broadcast doesn't work due to the bug in FPU tile
171  // which masks dest write enable signals when instrn_mode[1:0] == 2'b01
172  // ELTWADD with zeros will be used as a workaround
173  broadcast_type = p_elwise::SRCB_BCAST_COL;
174  }
175  else if constexpr (bcast_type == BroadcastType::ROW)
176  {
177  innerloop = (total_rows >> 3);
178  broadcast_type = p_movb2d::MOV_8_ROW_BRCST;
179  }
180  else if constexpr (bcast_type == BroadcastType::SCALAR)
181  {
182  // ELTWADD with zeros will be used as a workaround
183  outerloop = 1;
184  innerloop = num_faces * (total_rows >> 3);
185  broadcast_type = p_elwise::SRCB_BCAST_ALL;
186  }
187 
188  if constexpr (bcast_type == BroadcastType::SCALAR)
189  {
190  ckernel_template tmp(outerloop, innerloop, TT_OP_ELWADD(0, 0, broadcast_type, addr_mod, 0));
191  tmp.set_end_op(TT_OP_SETRWC(p_setrwc::CLR_AB, 0, 0, 0, 0, 0));
192  tmp.program(instrn_buffer);
193  }
194  else if constexpr (bcast_type == BroadcastType::COL)
195  {
196  ckernel_template tmp(outerloop, innerloop, TT_OP_ELWADD(0, 0, broadcast_type, addr_mod, 0));
197  tmp.set_end_op(TT_OP_SETRWC(0, p_setrwc::CR_B, 0, 0, 0, p_setrwc::SET_B));
198  tmp.program(instrn_buffer);
199  }
200  else if constexpr (bcast_type == BroadcastType::ROW)
201  {
202  ckernel_template tmp(outerloop, innerloop, TT_OP_MOVB2D(0, 0, addr_mod, broadcast_type, 0));
203  tmp.set_end_op(TT_OP_SETRWC(p_setrwc::CLR_B, p_setrwc::CR_B, 0, 0, 0, p_setrwc::SET_B));
204  tmp.program(instrn_buffer);
205  }
206  else
207  {
208  ckernel_template tmp(outerloop, innerloop, TT_OP_MOVB2D(0, 0, addr_mod, rows_per_inst, 0));
209  tmp.set_end_op(TT_OP_SETRWC(p_setrwc::CLR_B, p_setrwc::CR_B, 0, 0, 0, p_setrwc::SET_B));
210  tmp.program(instrn_buffer);
211  }
212  }
