40 inline void _llk_pack_untilize_mop_config_(
41  const std::uint32_t face_r_dim = FACE_R_DIM, const std::uint32_t num_faces = 4, bool narrow_row = false, std::uint32_t row_num_datums = TILE_C_DIM)
42 {
43  constexpr uint MEGAROW = 1;
44  constexpr uint ZERO_OUTPUT_FLAG = p_pacr::P_ZERO_OUTPUT_DISABLED;
45  constexpr uint MOP_INNER_LOOP = block_ct_dim;
46 
47  // Loop until face_r_dim - 1.
48  // Last row of face needs to be handled differently depending on num_faces, block_ct and full_ct.
49  const uint MOP_OUTER_LOOP = face_r_dim - 1;
50 
51  const uint PACK_INTF_SEL = (narrow_row) ? p_pacr::SINGLE_INTF_ACTIVE : ((num_faces > 1) ? p_pacr::TWO_INTFS_ACTIVE : p_pacr::SINGLE_INTF_ACTIVE);
52 
53  bool outer_loop_valid = (MOP_OUTER_LOOP > 0) && (MOP_OUTER_LOOP < 128);
54  bool inner_loop_valid = (MOP_INNER_LOOP > 0) && (MOP_INNER_LOOP < 128);
55  // Currently no way to check if MOP properly configured when issuing MOP instruction
56  // so we check here and guard with a default configuration that only has NOPs
57  if (outer_loop_valid && inner_loop_valid)
58  {
59  /*
60  Each pack instruction does 2x16 datums if (num_faces>1)
61  Each row of 16 datums, has a stride of 16 from dest read
62  Dest row read in inner loop:
63  tile 0: row 0, row 16
64  tile 1: row 64, row 80
65  .
66  tile block_ct_dim-1: row 64*(block_ct_dim-1), row 64*(block_ct_dim-1)+16
67  */
68 
69  ckernel::ckernel_template tmp(
70  MOP_OUTER_LOOP,
71  MOP_INNER_LOOP,
72  TT_OP_PACR(
73  p_pacr::CFG_CTXT_0,
74  p_pacr::NO_ROW_PAD_ZERO,
75  p_pacr::DST_ACCESS_STRIDED_MODE,
76  ADDR_MOD_0,
77  p_pacr::ADDR_CNT_CTXT_0,
78  0,
79  PACK_INTF_SEL,
80  0,
81  MEGAROW,
82  p_pacr::NO_CTXT_CTRL,
83  0,
84  0),
85  TT_OP_INCADCZW(p_setadc::PAC, 0, 0, 1, 0) // w cnt points to the next tile
86  );
87 
88  // reset ch0_w counters
89  tmp.set_start_op(TT_OP_SETADCZW(p_setadc::PAC, 0, 0, 0, 0, 0b0010));
90 
91  if constexpr (block_ct_dim != full_ct_dim)
92  {
93  const std::uint32_t replay_buf_len = 4;
94  load_replay_buf(
95  ckernel::packer::replay_buf_offset,
96  replay_buf_len,
97  false,
98  []
99  {
100  // update l1 address
101  TTI_ADDDMAREG(0, p_gpr_pack::OUTPUT_ADDR, p_gpr_pack::OUTPUT_ADDR, p_gpr_pack::OUTPUT_ADDR_OFFSET);
102  TTI_STALLWAIT(p_stall::STALL_CFG, p_stall::THCON);
103  TTI_WRCFG(p_gpr_pack::OUTPUT_ADDR, 0, THCON_SEC0_REG1_L1_Dest_addr_ADDR32);
104  TTI_NOP;
105  });
106 
107  tmp.set_end_ops(
108  TT_OP_INCADCXY(p_setadc::PAC, 0, 0, 1, 0), // inc ch0_y counters
109  TT_OP_REPLAY(ckernel::packer::replay_buf_offset, replay_buf_len, 0, 0) // update row address
110  );
111  }
112  else
113  {
114  tmp.set_end_op(TT_OP_INCADCXY(p_setadc::PAC, 0, 0, 1, 0) // inc ch0_y counters
115  );
116  }
117  tmp.program(instrn_buffer);
118  }
119  // If wanted MOP config is not valid, create a default one.
120  // This is due to not being able to check if MOP config is valid before issuing in runtime.
121  else
122  {
123  ckernel::ckernel_template tmp(1, 1);
124  tmp.program(instrn_buffer);
125  }
126 }

134 inline void _llk_pack_untilize_init_(
135  const std::uint32_t pack_src_format, const std::uint32_t pack_dst_format, const std::uint32_t face_r_dim = FACE_R_DIM, const std::uint32_t num_faces = 4)
136 {
137  static_assert(!diagonal, "Diagonal not supported");
138  _llk_pack_untilize_configure_addrmod_<diagonal>();
139 
140  _llk_pack_untilize_mop_config_<block_ct_dim, full_ct_dim, diagonal>(face_r_dim, num_faces, narrow_row, row_num_datums);
141 
142  // Set CH0 Zstride = 2x16x16 faces, .z_src = {.incr = 1} jumps 2 faces
143  uint x_stride = (uint)(pack_src_format & 0x3) == (uint)DataFormat::Float32 ? 4 : (uint)(pack_src_format & 0x3) == (uint)DataFormat::Float16 ? 2 : 1;
144  uint y_stride = FACE_C_DIM * x_stride;
145  const uint z_stride = 2 * face_r_dim * y_stride;
146  cfg_reg_rmw_tensix<PCK0_ADDR_CTRL_ZW_REG_0_Zstride_RMW>(z_stride);
147 
148  if (block_ct_dim != full_ct_dim)
149  {
150  const std::uint32_t output_addr_offset = SCALE_DATUM_SIZE(pack_dst_format, full_ct_dim * ((num_faces > 1) ? (num_faces >> 1) : 1) * FACE_C_DIM);
151  TT_SETDMAREG(0, LOWER_HALFWORD(output_addr_offset / 16), 0, LO_16(p_gpr_pack::OUTPUT_ADDR_OFFSET)); // store 16B aligned row offset address
152  }

176 
177  for (std::uint32_t face = 0; face < num_faces_per_rdim_tile; face++)
178  {
179  ckernel::ckernel_template::run(instrn_buffer);
180 
181  //-----------------------------------------------------------------------
182  // Handle last row of face, i.e. last outer_loop iteration of MOP.
183  // Start OP is the same for all cases.
184  TTI_SETADCZW(p_setadc::PAC, 0, 0, 0, 0, 0b0010); // reset ch0 W counter
185 
186  // Inner loop of MOP.
187  for (std::uint32_t i = 0; i < block_ct_dim; i++)
188  {
189  // Close block if it is the last PACR instruction of the block.

200  0,
201  0 /*MEGAROW*/,
202  p_pacr::NO_CTXT_CTRL,
203  0,
204  1);
205  }
206  else

208  TTI_PACR(
209  p_pacr::CFG_CTXT_0,
210  p_pacr::NO_ROW_PAD_ZERO,
211  p_pacr::DST_ACCESS_STRIDED_MODE,
212  ADDR_MOD_0,
213  p_pacr::ADDR_CNT_CTXT_0,
214  p_pacr::P_ZERO_OUTPUT_DISABLED,
215  PACK_INTF_SEL,
216  0,
217  1 /*MEGAROW*/,
218  p_pacr::NO_CTXT_CTRL,
219  0,
220  0);
221  }
222 
223  TTI_INCADCZW(p_setadc::PAC, 0, 0, 1, 0); // w cnt points to the next tile
224  }
225 
226  // End OP.
227  TTI_INCADCXY(p_setadc::PAC, 0, 0, 1, 0); // inc ch0_y counters
228  if (block_ct_dim != full_ct_dim)
229  {
230  // update l1 address
231  TTI_ADDDMAREG(0, p_gpr_pack::OUTPUT_ADDR, p_gpr_pack::OUTPUT_ADDR, p_gpr_pack::OUTPUT_ADDR_OFFSET);
232  TTI_STALLWAIT(p_stall::STALL_CFG, p_stall::THCON);
233  TTI_WRCFG(p_gpr_pack::OUTPUT_ADDR, 0, THCON_SEC0_REG1_L1_Dest_addr_ADDR32);
234  TTI_NOP;
235  }
236  //-----------------------------------------------------------------------
237 
238  TTI_INCADCZW(p_setadc::PAC, 0, 0, 0, 1); // z cnt increments by 2xface_r_dimxFACE_C_DIM
239  TTI_SETADCXY(p_setadc::PAC, 0, 0, 0, 0, 0b0010); // reset ch0_y counters
240  }
241 
242  TT_SETADCZW(p_setadc::PAC, 0, 0, 0, 0, 0b0101); // reset z counters
