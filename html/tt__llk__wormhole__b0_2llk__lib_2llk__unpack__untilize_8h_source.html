86  cfg_reg_rmw_tensix<THCON_SEC0_REG0_TileDescriptor_ADDR32 + 1, 0, 0xFFFF>(FACE_C_DIM);
87  TTI_REG2FLOP(
88  1, 0, 0, 0, THCON_SEC0_REG5_Tile_x_dim_cntx0_ADDR32 - THCON_CFGREG_BASE_ADDR32, p_gpr_unpack::FACE_DIM_1x16); // GPR preloaded with 16 | (16 << 16)
89 
90  TT_SETDMAREG(0, LOWER_HALFWORD(tile_size), 0, LO_16(p_gpr_unpack::TILE_SIZE));
91  TT_SETDMAREG(0, UPPER_HALFWORD(tile_size), 0, HI_16(p_gpr_unpack::TILE_SIZE));
92 
93  _llk_unpack_untilize_mop_config_();
94 }

96 template <bool first_pass = true>
97 inline void _llk_unpack_untilize_pass_(const std::uint32_t base_address, const std::uint32_t block_tile_cols)
98 {
99  std::uint32_t rem_blocks_in_row = block_tile_cols;
100 
101  // Program srcA and srcB base addresses
102  volatile uint tt_reg_ptr *cfg = get_cfg_pointer(); // get pointer to registers for current state ID
103 
104  TTI_SETADCXY(0b001, 0, 0, 0, 0, 0b0010); // Clear l1 addr y cnt
105  if constexpr (first_pass)
106  {
107  // Select bootom faces in the 2nd pass
108  TT_SETADC(p_setadc::UNP0, p_setadc::CH_0, p_setadc::SET_Z, 0);
109  }
110  else
111  {
112  // Select bootom faces in the 2nd pass
113  TT_SETADC(p_setadc::UNP0, p_setadc::CH_0, p_setadc::SET_Z, 2);
114  }
115 
116  // Wait for free context
117  wait_for_next_context(2);
118 
119  // Get tile address
120  if (0 == unp_cfg_context)
121  {
122  cfg[THCON_SEC0_REG3_Base_address_ADDR32] = base_address;
123  }
124  else
125  {
126  cfg[THCON_SEC0_REG3_Base_cntx1_address_ADDR32] = base_address;
127  }
128 
129  // Trisc::SEMPOST for context acquire
130  semaphore_post(semaphore::UNPACK_SYNC);
131 
132  // Stall unpacker until pending CFG writes from Trisc have completed
133  TTI_STALLWAIT(p_stall::STALL_UNPACK, p_stall::TRISC_CFG);
134 
135  std::uint32_t face_2xr_cnt = 0;
136  for (std::uint32_t r = 0; r < FACE_HEIGHT; r++)
137  {
138  rem_blocks_in_row = block_tile_cols; // reset remaining blocks in row
139 
140  do
141  {
142  if ((face_2xr_cnt + rem_blocks_in_row) >= (FACE_HEIGHT / 2))
143  {
144  // Run MOP
145  TT_MOP(0, 8 - face_2xr_cnt - 1, unp_cfg_context == 0 ? 0 : 0xff); // Run the MOP
146 #if SKIP_UNP == 1
147  TTI_NOP;
148 #else
149  TTI_UNPACR(SrcA, 0b0, 0, 0, 0, 1, 1, p_unpacr::RAREFYB_DISABLE, 0, 0, 0, 0, 1); // set data valid
150  TTI_UNPACR_NOP(SrcB, p_unpacr_nop::UNP_ZEROSRC);
151  TTI_UNPACR_NOP(SrcB, p_unpacr_nop::UNP_SET_DVALID);
152 #endif
153  TTI_SETADCXY(0b001, 0, 0, 0, 0, 0b1000); // Clear srcA addr y cnt
154  rem_blocks_in_row -= (8 - face_2xr_cnt);
155  face_2xr_cnt = 0;
156  }
157  else
158  {
159  TT_MOP(0, rem_blocks_in_row - 1, unp_cfg_context == 0 ? 0 : 0xff); // Run the MOP
160  face_2xr_cnt += rem_blocks_in_row;
161  rem_blocks_in_row = 0;
162  // if (face_2xr_cnt==FACE_HEIGHT/2) {
163  // TTI_UNPACR(SrcA, 0b0, 0, 0, 0, 0, 1, p_unpacr::RAREFYB_DISABLE, 0, 0, 0, 0, 1); //set data valid
164  // TTI_SETADCXY(0b001, 0, 0, 0, 0, 0b1000); // Clear srcA addr y cnt
165  // face_2xr_cnt = 0;
166  //}
167  }
168  } while (rem_blocks_in_row > 0);
169 
170  TTI_MULDMAREG(0, p_gpr_unpack::TILE_OFFSET, p_gpr_unpack::TILE_OFFSET, p_gpr::ZERO); // TILE_OFFSET=TILE_OFFSET*0
171  if (0 == unp_cfg_context)
172  {
173  TTI_REG2FLOP(1, 0, 0, 0, THCON_SEC0_REG7_Offset_address_ADDR32 - THCON_CFGREG_BASE_ADDR32,
174  p_gpr::ZERO); // Clear offset register
175  }
176  else
177  {
178  TTI_REG2FLOP(1, 0, 0, 0, THCON_SEC0_REG7_Offset_cntx1_address_ADDR32 - THCON_CFGREG_BASE_ADDR32,
179  p_gpr::ZERO); // Clear offset register
180  }
181  TTI_INCADCXY(0b001, 0, 0, 1, 0); // inc l1 addr y cnt
182  }
183 
184  // T6::SEMGET for context release
185  t6_semaphore_get(semaphore::UNPACK_SYNC);
186 
187  // Switch unpacker config context
188  switch_config_context(unp_cfg_context);
189 
190 #ifdef PERF_DUMP
191  first_unpack_recorded = true;
192 #endif
193 }
