15 sfpi_inline sfpi::vFloat _sfpu_sine_maclaurin_series_(sfpi::vFloat val)
16 {
17  // Good for [-pi:pi]
18  // Mclauren series = x - x^3/3! + x^5/5! - x^7/7! + x^9/9! - x^11/11!
19  sfpi::vFloat tmp = val;
20  // x
21  sfpi::vFloat output = tmp;
22  // x^3/3!
23  tmp = tmp * val * val;
24  output += -0.166666666 * tmp;
25  // x^5/5!
26  tmp = tmp * val * val;
27  output += 0.0083333333 * tmp;
28  // x^7/7!
29  tmp = tmp * val * val;

35  output += 0.0000027557 * tmp;
36  // x^11/11!
37  tmp = tmp * val * val;

75 template <bool APPROXIMATION_MODE, int ITERATIONS>
76 inline void _calculate_sine_(const int iterations)
77 {
78  // SFPU microcode
79  for (int d = 0; d < iterations; d++)
80  {
81  sfpi::vFloat v = sfpi::dst_reg[0];
82  v = 0.318309886183791f * v; // *1/pi to get number of pi rads.
83  sfpi::vInt whole_v = float_to_int16(v, 0);
84  sfpi::vFloat whole_v_float = int32_to_float(whole_v, 0);
85  v = v - whole_v_float;
86  v *= 3.141592653589793f; // fractional * pi to get it in [-pi:pi]
87  v = _sfpu_sine_maclaurin_series_<APPROXIMATION_MODE>(v);
88  whole_v = whole_v & 0x1;
89  v_if (whole_v != 0)
90  {
91  // odd so flip the sign
92  v *= -1;
93  }
94  v_endif;
95  sfpi::dst_reg[0] = v;
96  sfpi::dst_reg++;
97  }
98 }
