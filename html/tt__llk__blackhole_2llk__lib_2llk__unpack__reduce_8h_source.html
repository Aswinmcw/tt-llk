87 template <PoolType type, ReduceDim dim>
88 inline void _llk_unpack_reduce_(const std::uint32_t address)
89 {
90  // Clear z/w start counters
91  TTI_SETADCZW(0b011, 0, 0, 0, 0, 0b1111);
92 
93  // Program srcA and srcB base addresses
94  volatile uint tt_reg_ptr *cfg = get_cfg_pointer(); // get pointer to registers for current state ID
95 
96  // Wait for free context
97  wait_for_next_context(2);
98 
99  // Get tile address
100  if (0 == unp_cfg_context)
101  {
102  cfg[THCON_SEC0_REG3_Base_address_ADDR32] = address;
103  }
104  else
105  {
106  cfg[THCON_SEC0_REG3_Base_cntx1_address_ADDR32] = address;
107  }
108 
109  // Trisc::SEMPOST for context acquire
110  semaphore_post(semaphore::UNPACK_SYNC);
111 
112  // Load only 16 datums into srcB
113  TTI_SETADCXX(p_setadc::UNP1, FACE_C_DIM - 1, 0x0);
114 
115  // Stall unpacker until pending CFG writes from Trisc have completed
116  TTI_STALLWAIT(p_stall::STALL_UNPACK, p_stall::TRISC_CFG);
117 
118  // Run MOP
119  mop_run(0, 4);
120 
121  // Restore face height
122  TTI_SETADCXX(p_setadc::UNP1, FACE_R_DIM * FACE_C_DIM - 1, 0x0);
123 
124  // T6::SEMGET for context release
125  t6_semaphore_get(semaphore::UNPACK_SYNC);
126 
127  // Switch unpacker config context
128  switch_config_context(unp_cfg_context);
129 
130 #ifdef PERF_DUMP
131  first_unpack_recorded = true;
132 #endif
133 }
