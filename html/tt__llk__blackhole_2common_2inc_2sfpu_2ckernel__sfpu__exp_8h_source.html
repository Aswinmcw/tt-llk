18 sfpi_inline sfpi::vFloat _sfpu_exp_(sfpi::vFloat val)
19 {
20  // If exponent is > -1 extract it and replace with -1
21  sfpi::vInt exp = exexp(val);
22  v_if (exp >= 0)
23  {
24  val = setexp(val, 126);
25  }
26  v_endif;
27 
28  // Run series in Horner form
29  sfpi::vFloat tmp = val * sfpi::vConst0p8373 + sfpi::s2vFloat16b(0.863281);
30  val = val * tmp + sfpi::vConst1;
31 
32  v_if (exp >= 0)
33  {
34  val = val * val;
35  for (int s_iter = 0; s_iter < 7; s_iter++)
36  {
37  exp = exp - 1;
38  // Narrow predication on each loop
39  v_and(exp >= 0);
40  val = val * val;
41  }
42  }
43  v_endif;
44 
45  return val;
46 }

48 template <bool APPROXIMATION_MODE>
49 sfpi_inline sfpi::vFloat _calculate_exponential_body_(sfpi::vFloat in)
50 {
51  sfpi::vFloat out;
52 
53  if constexpr (APPROXIMATION_MODE)
54  {
55  constexpr int FRAC_BITS = 3;
56  constexpr uint SP_BIAS = 127 << FRAC_BITS;
57 
58  // * by 1/ln2 and add convert to 7.3 FxP format
59  sfpi::vFloat vConstLn2Recip = sfpi::vConstFloatPrgm0;
60  sfpi::vFloat conv = in * vConstLn2Recip;
61 
62  // Clear exp bits
63  sfpi::vInt c23_73 = p_exp::C23_73;
64  sfpi::vInt tmp = sfpi::reinterpret<sfpi::vInt>(conv) - c23_73;
65 
66  // Add bias
67  tmp += SP_BIAS;
68 
69  // SHL to move integer bits to exponent
70  out = sfpi::reinterpret<sfpi::vFloat>(tmp << (10 - FRAC_BITS));
71  }
72  else
73  {
74  // Force sign to 0 (make number positive)
75  out = _sfpu_exp_(sfpi::setsgn(in, 0));
76 
77  v_if (in < 0)
78  {
79  out = _sfpu_reciprocal_(out);
80  }
81  v_endif;
82  }
83 
84  return out;
85 }

91  {
92  // Sanitize the input values by loading from DEST, comparing against the value -88.5, and if the input value is more negative than that, swap the input
93  // value with -88.5 and store back to DEST
94  // - in other words, after the sanitize step, the values in DEST will be in the range {-88.5 , +inf}
95 
96  // Macro Sequence Register 1 configured to read back in the original values from dest, sanitize them to a range we can handle, and then store them back
97  // to dest
98  // LD : bring in the original value from DEST (y)
99  // MAD : unused
100  // ROUND : unused
101  // SIMPLE : SWAP the larger value of y and -88.5 into the LREG
102  // STORE : store the sanitized value back to dest
103  TTI_SFPLOADMACRO(
104  4,
105  0,
106  ADDR_MOD_7,
107  0); // MACRO Sequence Register 1: LD, SWAP, STORE - uses LREG[0] for loaded value - Dest offset 0 is targeting the even columns for rows 3: 0
108  TTI_SFPNOP; // NOP is necessary because the SWAP operation takes 2 cycles and unfortunately is not pipelined
109  TTI_SFPLOADMACRO(
110  5,
111  0,
112  ADDR_MOD_7,
113  2); // MACRO Sequence Register 1: LD, SWAP, STORE - uses LREG[1] for loaded value - Dest offset 2 is targeting the odd columns for rows 3: 0
114  TTI_SFPNOP;
115  TTI_SFPLOADMACRO(
116  6,
117  0,
118  ADDR_MOD_7,
119  4); // MACRO Sequence Register 1: LD, SWAP, STORE - uses LREG[2] for loaded value - Dest offset 4 is targeting the even columns for rows 7: 4
120  TTI_SFPNOP;
121  TTI_SFPLOADMACRO(
122  7,
123  0,
124  ADDR_MOD_7,
125  6); // MACRO Sequence Register 1: LD, SWAP, STORE - uses LREG[3] for loaded value - Dest offset 6 is targeting the odd columns for rows 7: 4
126  TTI_SFPNOP;
127  TTI_SFPLOADMACRO(
128  4,
129  0,
130  ADDR_MOD_7,
131  8); // MACRO Sequence Register 1: LD, SWAP, STORE - uses LREG[0] for loaded value - Dest offset 8 is targeting the even columns for rows 11: 8
132  TTI_SFPNOP;

139  TTI_SFPLOADMACRO(

144  TTI_SFPNOP;
145  TTI_SFPLOADMACRO(
146  7,
147  0,
148  ADDR_MOD_7,
149  14); // MACRO Sequence Register 1: LD, SWAP, STORE - uses LREG[3] for loaded value - Dest offset 14 is targeting the even columns for rows 15:12
150  // NOP not needed in this spot because the next LoadMacro is a computational macro which doesn't immediately use the SIMPLE unit
151 
152  // Macro Sequence Register 0 configured to read back in the sanitized values and calculate the approximate exponential value
153  // LD : the sanitized value from DEST (y)
154  // MAD : compute (A * y) + (B-C) , where A = (2^8)/ln(2) , B = 127 * (2^8) , C = Adjustment parameter of roughly 11.2 to minimize error
155  // ROUND : convert the MAD result from FP32 to a 16-bit unsigned integer using stochastic rounding
156  // SIMPLE : shift the 16-bit integer to the left by 15 bits to place the MSB of the computed value into the MSB of the exponent bits of the fp32 format
157  // STORE : store the shifted value back to dest
158  TTI_SFPLOADMACRO(0, 0, ADDR_MOD_7, 0); // MACRO Sequence Register 0: LD, MAD, ROUND, SHIFT and STORE - uses LREG[0] for loading and intermediate results
159  // - Dest offset 0 is targeting the even columns for rows 3: 0
160  TTI_SFPLOADMACRO(1, 0, ADDR_MOD_7, 2); // MACRO Sequence Register 0: LD, MAD, ROUND, SHIFT and STORE - uses LREG[1] for loading and intermediate results
161  // - Dest offset 2 is targeting the odd columns for rows 3: 0
162  TTI_SFPLOADMACRO(2, 0, ADDR_MOD_7, 4); // MACRO Sequence Register 0: LD, MAD, ROUND, SHIFT and STORE - uses LREG[2] for loading and intermediate results
163  // - Dest offset 4 is targeting the even columns for rows 7: 4
164  TTI_SFPLOADMACRO(3, 0, ADDR_MOD_7, 6); // MACRO Sequence Register 0: LD, MAD, ROUND, SHIFT and STORE - uses LREG[3] for loading and intermediate results
165  // - Dest offset 6 is targeting the odd columns for rows 7: 4
166  TTI_SFPLOADMACRO(0, 0, ADDR_MOD_7, 8); // MACRO Sequence Register 0: LD, MAD, ROUND, SHIFT and STORE - uses LREG[0] for loading and intermediate results
167  // - Dest offset 8 is targeting the even columns for rows 11: 8
168  TTI_SFPLOADMACRO(1, 0, ADDR_MOD_7, 10); // MACRO Sequence Register 0: LD, MAD, ROUND, SHIFT and STORE - uses LREG[1] for loading and intermediate
169  // results - Dest offset 10 is targeting the even columns for rows 11: 8
170  TTI_SFPLOADMACRO(2, 0, ADDR_MOD_7, 12); // MACRO Sequence Register 0: LD, MAD, ROUND, SHIFT and STORE - uses LREG[2] for loading and intermediate
171  // results - Dest offset 12 is targeting the odd columns for rows 15:12
172  TTI_SFPLOADMACRO(3, 0, ADDR_MOD_7, 14); // MACRO Sequence Register 0: LD, MAD, ROUND, SHIFT and STORE - uses LREG[3] for loading and intermediate
173  // results - Dest offset 14 is targeting the even columns for rows 15:12
174  // NOP needed to allow time for the final Computation Loadmacro to complete before returning to the Sanitation Loadmacro at the top for the next
175  // iteration
176  // - to be completely safe, use 3 NOP; in practice 1 seems to be enough, probably because the overhead of the DEST INCRW stuff introduces 2 cycles of
177  // delay
178  TTI_SFPNOP;
179  // TTI_SFPNOP;
180  // TTI_SFPNOP;
181  }

188  if constexpr (SCALE_EN)

190  val = val * sfpi::s2vFloat16a(exp_base_scale_factor);
191  }
192  if constexpr (APPROXIMATION_MODE)
193  {
194  v_if (val >= 89)

198  }

202  }
203  v_else

205  // * by 1/ln2 and add convert to 7.3 FxP format

208  sfpi::vInt adj_exp = sfpi::vConstIntPrgm2;
209  val = val * vConstLn2Recip + c23_73;
210 
211  // Remove Exponent of 7 and bias the Mantissa to 127.
212  sfpi::vInt val_short = adj_exp + sfpi::reinterpret<sfpi::vInt>(val);
213 
214  // SHL to move integer bits to exponent
215  val_short <<= 10 - p_exp::FRAC_BITS;
216  sfpi::dst_reg[0] = sfpi::reinterpret<sfpi::vFloat>(val_short);
217  }
218  v_endif;
219  }
220  else
221  {
222  // Force sign to 0 (make number positive)
223  sfpi::vFloat result = _sfpu_exp_(sfpi::setsgn(val, 0));
224 
225  v_if (val < 0)
226  {
227  result = _sfpu_reciprocal_(result);
228  }
229  v_endif;
230 
231  sfpi::dst_reg[0] = result;
232  }
233 
234  sfpi::dst_reg++;
235  }
236  }

239 template <bool APPROXIMATION_MODE, bool FAST_APPROX = true>
240 inline void _init_exponential_()
241 {
242  if constexpr (FAST_APPROX && APPROXIMATION_MODE)
243  {
244  // Algorithm is adapted from:
245  // A Fast, Compact Approximation of the Exponential Function
246  // Nicol N. Schraudolph
247  // IDSIA, Lugano, Switzerland
248 
249  // First, set up constant values which are needed for the computation
250  // We will first sanitize the input values (y) to be in the range that won't cause underflow, which for our hardware means we need to limit
251  // negative values to be greater than or equal to -88.5 The computation that is needed is (A * y) + (B - C) , where A = (2^8)/ln(2) , B = 127 *
252  // (2^8) , C = Adjustment parameter of roughly 11.2 to minimize error
253  // - NOTE: we would like to be able to use 2^23 instead of 2^8 and compute a 32-bit quantity, but our hardware only supports rounding FP32 into
254  // a 16-bit integer, so we use 2^8 and then shift left by 15 bits after rounding
255  // So we will set up the following constants:
256  // LREG[14] = = -88.5 = 0xc2b10000
257  // LREG[12] = A = 369.329925537109375 = 0x43b8aa3b
258  // LREG[13] = (B-C) = 32500.818359375 = 0x46fde9a3
259 
260  TTI_SFPLOADI(0, 0xA, 0x0000);
261  TTI_SFPLOADI(0, 0x8, 0xC2B1);
262  TTI_SFPCONFIG(0, 14, 0); // SFPCONFIG Dest 14 = LREG[14] = -88.5 = 0xc2b10000
263 
264  TTI_SFPLOADI(0, 0xA, 0xaa3b);
265  TTI_SFPLOADI(0, 0x8, 0x43B8);
266  TTI_SFPCONFIG(0, 12, 0); // SFPCONFIG Dest 12 = LREG[12] = A = 369.329925537109375 = 0x43b8aa3b
267 
268  TTI_SFPLOADI(0, 0xA, 0xe9a3);
269  TTI_SFPLOADI(0, 0x8, 0x46Fd);
270  TTI_SFPCONFIG(0, 13, 0); // SFPCONFIG Dest 13 = LREG[13] = (B-C) = 32500.818359375 = 0x46fde9a3
271 
272  // Next, set up the macro instructions which will be necessary
273  // - for the sanitize function: we will need a SWAP instruction
274  // - for the main computation function: we will need MAD, ROUND, and SHIFT instructions
275 
276  // There are two ways to program the macro instruction registers, and this setup leverages both ways
277  // - we can either use the SFPCONFIG flow, by setting up the bits of the instruction into LREG[0] and then targeting the Macro instruction register
278  // - or we can use the shortcut / backdoor load method which relies on having some illegal destination register values as part of the instruction
279 
280  // Use SFPCONFIG method for the SWAP instruction, since we want the SWAP itself to use a destination register which is not normally a legal value
281  // (we are cheating a bit here, since we only care about one half of the swap and we want to use a constant for the other half)
282  //
283  // imm12 = 0, lreg_src_c = 0 (will be fed by value loaded from Dest into Loadmacro lreg_dest), lreg_dest = LREG[14] = - 88.5,
284  // instr_mod1 = 1 swap the values with the larger of the two ending up in lreg_dest -> but we will use the Loadmacro lreg_dest register as
285  // output
286  // TTI_SFP_SWAP(0, 0, 14, 1);
287  TTI_SFPLOADI(0, 0xA, 0x00E1);
288  TTI_SFPLOADI(0, 0x8, 0x9200);
289  TTI_SFPCONFIG(0, 0, 0); // SFPCONFIG Dest 0 = Programmable Macro instruction 0: TTI_SFPSWAP(0, 0, 14, 1); // compare against LREG[14] (-88.5), and put
290  // the larger value into LREG[loadmacro_lreg_dest]
291  TTI_SFPNOP;
292 
293  // Backdoor load of Macro Instruction 1
294  // Dummy version of MAD instruction with lreg_dest = 4'b11_01 = 13 to install into Programmable Macro instruction register 1, which is Macro Instruction
295  // Register 5
296  TTI_SFPMAD(12, 0, 13, 13, 0); // MACRO Instruction 1 <--- lreg X = lreg[12] (A) * lreg[0] (y) + lreg[13] (B-C)
297 
298  // Backdoor load of Macro Instruction 2
299  // ROUND instruction to convert FP32 result into an integer value (int16)
300  // Stochastic = 0, Imm(Descale), SrcB(unused), SrcC(input value), Lreg_dest = 14 to install in Programmable Macro Instruction reg
301  // 2'b10, instr_mod1 = 14 to treat input as fp32, output as unsigned int16, use imm as descale
302  TTI_SFP_STOCH_RND(0, 0, 0, 0, 14, 14); // Round to unsigned Int16
303 
304  // Backdoor load of Macro Instruction 3
305  // If using the unsigned int rounding mode, then shift by 15; SHL to move integer bits to exponent;
306  TTI_SFPSHFT(15, 0, 15, 1); // imm = 15 to shift left by 15 bits; lreg_c = 0 (will use macro reg); lreg_dest = 15 to install in Programmable Macro
307  // Instruction reg 2'b11, which is Macro Instruction Register 7
308 
309  // So at this point, we have the following instructions loaded into our macro registers:
310  //
311  // 00: (no macro instruction, just execute whatever is issued from Tensix) <-- these are fixed / not programmable
312  // 01: ( Rsvd ) <-- these are fixed / not programmable
313  // 02: ( NOP ) <-- these are fixed / not programmable
314  // 03: ( SFPSTORE ) <-- these are fixed / not programmable
315  // 04: TTI_SFPSWAP (0, 0, 11, 1)
316  // 05: TTI_SFPMAD (12, 0, 13, 13, 0)
317  // 06: TTI_SFP_STOCH_RND (1, 0, 0, 0, 14, 14)
318  // 07: TTI_SFPSHFT (15,0,15,1)
319 
320  // Now we want to set up our two sequences
321 
322  // Sequence 1 setup: we want to Load, SWAP, <delay>, Store
323  // Delay slot: 0 1 2
324  // Use
325  // Loaded Result Macro
326  // Value Value Delay Instruction
327  // SRCB Stage Slot Select
328  TTI_SFPLOADI(0, 0xA, 0x0004); // slot1 : SIMPLE UNIT, want SWAP instruction which is in macro instruction mux[4], delayed by 0 ; not using staging flop
329  // as dest; not using load reg as srcb : 8'b0_______0_______000_____100 = 0x04 slot2 : MAD UNIT, unused :
330  // 8'b0_______0_______000_____000 = 0x00
331  TTI_SFPLOADI(0, 0x8, 0x1300); // slot3 : ROUND UNIT, unused : 8'b0_______0_______000_____000 = 0x00 slot4 : STORE UNIT, want STORE
332  // instruction which is in macro instruction mux[3], delayed by 2 ; not using staging flop as src ; :
333  // 8'b0_______0_______010_____011 = 0x13
334  TTI_SFPCONFIG(0, 5, 0); // SFPCONFIG Dest 5 = Macro Sequence Register 1
335 
336  // Sequence 0 setup: we want to Load, MAD, <delay>, ROUND, SHIFT, Store
337  // Delay slot: 0 1 2 3 4
338  // Use
339  // Loaded Result Macro
340  // Value Value Delay Instruction
341  // SRCB Stage Slot Select
342  TTI_SFPLOADI(
343  0,
344  0xA,
345  0x85DF); // slot1 : SIMPLE UNIT, want SHIFT instruction which is in macro instruction mux[7], delayed by 3 ; using staging flop as dest; using
346  // load reg as srcb : 8'b1_______1_______011_____111 = 0xDF slot2 : MAD UNIT, want MAD instruction which is in macro
347  // instruction mux[5], delayed by 0 ; not using staging flop as dest; using load reg as srcb : 8'b1_______0_______000_____101 = 0x85
348  TTI_SFPLOADI(
349  0,
350  0x8,
351  0x6316); // slot3 : ROUND UNIT, want ROUND instruction which is in macro instruction mux[6], delayed by 2 ; not using staging flop as dest; using
352  // : 8'b0_______0_______010_____110 = 0x16 slot4 : STORE UNIT, want STORE instruction which is in macro instruction mux[3],
353  // delayed by 4 ; using staging flop as src ; using : 8'b0_______1_______100_____011 = 0x63
354  TTI_SFPCONFIG(0, 4, 0); // Load it into macro sequence register 0 (destination = 4)
355  }
356  else if constexpr (APPROXIMATION_MODE)
357  {
358  sfpi::vConstFloatPrgm0 = 1.442695f; // ln2_recip
359  sfpi::vConstFloatPrgm1 = sfpi::s2vFloat16b(p_exp::C23_73);
360  sfpi::vConstFloatPrgm2 = sfpi::s2vFloat16b(p_exp::ADJ_EXP);
361  }
362  else
363  {
364  sfpi::vConstFloatPrgm0 = 1.442695f; // ln2_recip
365  sfpi::vConstFloatPrgm1 = 2.0f;
366  sfpi::vConstFloatPrgm2 = 0.863281f;
367  }
368 }
